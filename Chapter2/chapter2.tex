\chapter{Khảo sát và phân tích các giải pháp liên quan}
\label{ch:cong-trinh-lien-quan}

\section{Các hệ thống quản lý tài chính cá nhân hiện có và hạn chế}
Trong quá trình xây dựng giải pháp quản lý tài chính cá nhân hiện nay, chúng tôi tiến hành khảo sát các ứng dụng phổ biến trên thị trường toàn cầu và Việt Nam, tiêu biểu như Money Lover, Spendee, và Wallet by BudgetBakers. Các phần mềm này đã giải quyết tốt một số nhu cầu cơ bản của người dùng, cụ thể:
\begin{itemize}
    \item \textbf{Money Lover:} Đây là ứng dụng quản lý tài chính nổi tiếng tại Việt Nam, hỗ trợ đắc lực việc ghi chép các khoản chi tiêu thường ngày, lập ngân sách định kỳ và quét hóa đơn thông qua ảnh chụp. Ứng dụng cung cấp các biểu đồ thống kê trực quan và hỗ trợ liên kết tài khoản ngân hàng để tự động cập nhật số dư.
    \item \textbf{Spendee:} Ứng dụng sở hữu thiết kế giao diện hiện đại, tối giản và tập trung mạnh vào các ví dùng chung (Shared Wallets), hỗ trợ gia đình hoặc các nhóm bạn bè cùng quản lý ngân sách chi tiêu.
    \item \textbf{Wallet by BudgetBakers:} Cung cấp giải pháp báo cáo chuyên sâu, hỗ trợ nhiều loại tiền tệ và tích hợp tính năng đồng bộ hóa tự động với hơn 4000 ngân hàng trên thế giới.
\end{itemize}

\textbf{Tuy nhiên, các hệ thống này vẫn tồn tại những hạn chế cốt lõi sau đối với đối tượng sinh viên:}
\begin{enumerate}
    \item \textbf{Thiếu tích hợp sâu mô hình 6 lọ:} Mặc dù người dùng có thể tự tạo các danh mục chi tiêu, các ứng dụng trên chưa số hóa quy chuẩn mô hình "6 cái lọ" của T. Harv Eker \cite{eker2005-6jars}. Người dùng phải tự thực hiện việc tính toán phần trăm phân bổ và chia nhỏ dòng tiền một cách thủ công mỗi khi có nguồn thu nhập mới.
    \item \textbf{Mức độ tự động hóa bằng AI còn thấp:} Đa số các ứng dụng chỉ sử dụng các thuật toán so khớp chuỗi (string matching) hoặc biểu thức chính quy (regular expressions) đơn giản để phân loại giao dịch. Khi người dùng nhập mô tả bằng ngôn ngữ tự nhiên viết tắt hoặc sai chính tả, hệ thống thường không nhận diện được và bắt buộc người dùng chọn danh mục thủ công, gây nản lòng và bỏ dở thói quen ghi chép.
    \item \textbf{Thiếu tính chủ động và tính cá nhân hóa (Proactivity):} Các ứng dụng hiện nay hoạt động theo mô hình thụ động (Passive Model), nghĩa là chỉ ghi nhận và thống kê số liệu sau khi giao dịch đã xảy ra. Hệ thống thiếu các cơ chế phân tích hành vi ngầm để đưa ra các gợi ý tái cấu trúc dòng tiền (Rebalancing) hoặc các cảnh báo chi tiêu vượt hạn mức một cách thông minh trước khi sự cố tài chính xảy ra.
\end{enumerate}

\section{Các giải pháp kết nối học bổng truyền thống}
Đối với sinh viên Việt Nam, các nguồn thông tin học bổng thường được công bố qua các kênh truyền thống như:
\begin{itemize}
    \item Trang web chính thức của Phòng Công tác Sinh viên (CTSV) hoặc Giáo vụ khoa của các trường Đại học.
    \item Các diễn đàn trực tuyến của sinh viên, các trang mạng xã hội (Facebook, LinkedIn).
    \item Các trang tin học bổng chuyên biệt (ví dụ: Scholarship Planet, YBOX).
\end{itemize}

\textbf{Các phương thức truyền thống này bộc lộ nhiều hạn chế rõ rệt:}
\begin{enumerate}
    \item \textbf{Thông tin bị phân mảnh và chậm trễ:} Sinh viên phải truy cập thủ công nhiều trang web khác nhau để tìm kiếm cơ hội. Nhiều học bổng có thời gian ứng tuyển ngắn dẫn đến việc sinh viên chỉ phát hiện ra khi đã sát hạn chót nộp hồ sơ.
    \item \textbf{Cơ chế lọc thông tin thô sơ:} Các trang tin học bổng hiện tại chỉ hỗ trợ lọc theo các tiêu chí cứng cơ bản (như Quốc gia, Ngành học chung). Sinh viên không thể biết chính xác hồ sơ năng lực hiện tại của mình (GPA, điểm rèn luyện, chứng chỉ tiếng Anh, hoạt động ngoại khóa) đáp ứng được bao nhiêu phần trăm điều kiện của học bổng đó.
    \item \textbf{Gánh nặng cho nhà tài trợ và ban quản trị:} Quy trình xét duyệt hồ sơ ứng tuyển của sinh viên đa phần vẫn thực hiện thủ công qua email hoặc Google Forms. Việc này gây khó khăn trong việc quản lý tập trung hồ sơ, dễ xảy ra sai sót khi nhập liệu và khó phát hiện các hành vi gian lận thông tin giữa các ứng viên.
\end{enumerate}

\section{Các giải pháp quản lý học thuật và bảng điểm hiện có}
Song song với nhu cầu quản lý tài chính, sinh viên còn cần theo dõi kết quả học tập và tiến độ tích lũy tín chỉ của bản thân. Các giải pháp hiện có cho nhu cầu này thường thuộc một trong hai nhóm:
\begin{itemize}
    \item \textbf{Cổng thông tin sinh viên (Student Portal) của từng trường đại học:} Đây là kênh chính thức để sinh viên tra cứu điểm số sau khi trường đã hoàn tất nhập liệu và công bố kết quả. Cổng thông tin đảm bảo tính chính xác tuyệt đối của dữ liệu điểm nhưng thường chỉ hiển thị kết quả cuối cùng, không hỗ trợ sinh viên tự nhập điểm dự kiến để mô phỏng (what-if) các kịch bản GPA trước khi có kết quả chính thức, và hoàn toàn tách biệt với các nhu cầu quản lý khác của sinh viên như tài chính hay học bổng.
    \item \textbf{Ứng dụng tính GPA độc lập:} Nhiều ứng dụng di động hoặc bảng tính (Excel/Google Sheets) cho phép sinh viên tự nhập điểm và số tín chỉ để tính GPA tích lũy. Tuy nhiên, các công cụ này không đồng bộ với danh mục môn học và thang điểm chuẩn hóa của từng trường, khiến sinh viên phải tự tra cứu và nhập thủ công số tín chỉ, dễ dẫn đến sai lệch khi trường có môn học trùng tên nhưng khác số tín chỉ giữa các khóa, hoặc thay đổi thang điểm quy đổi.
\end{itemize}

\textbf{Cả hai nhóm giải pháp trên đều tồn tại những hạn chế chung sau đối với sinh viên:}
\begin{enumerate}
    \item \textbf{Thiếu liên kết giữa dữ liệu học thuật và các nhu cầu khác của sinh viên:} Kết quả học tập (GPA, tiến độ tín chỉ) là một tiêu chí xét duyệt quan trọng của nhiều chương trình học bổng, nhưng sinh viên phải tự tra cứu, tổng hợp thủ công từ cổng thông tin trường để điền vào hồ sơ ứng tuyển học bổng ở một hệ thống khác, không có cơ chế đối chiếu tự động.
    \item \textbf{Không chuẩn hóa dữ liệu môn học giữa các trường:} Do mỗi trường có mã môn, số tín chỉ và thang điểm quy đổi riêng, các ứng dụng tính GPA độc lập không thể tự động xác thực dữ liệu sinh viên nhập vào, dẫn đến sai số tích lũy nếu sinh viên nhầm lẫn thông tin môn học.
\end{enumerate}

\section{Ứng dụng AI trong các hệ thống hỗ trợ sinh viên}
Sự bùng nổ của các Mô hình ngôn ngữ lớn (Large Language Models - LLM) và các hệ thống đại lý thông minh (AI Agents) đã mở ra hướng giải quyết mới cho các bài toán quản lý và gợi ý thông tin.

\subsection{Trong quản lý tài chính cá nhân}
Việc ứng dụng AI giúp chuyển dịch mô hình quản lý tài chính từ thụ động sang chủ động:
\begin{itemize}
    \item \textbf{Xử lý ngôn ngữ tự nhiên (NLP \cite{ddien-nlp-2006}):} Sử dụng LLM để hiểu ý định của người dùng qua tin nhắn thoại hoặc văn bản mô tả ngắn gọn. AI có khả năng phân tích ngữ cảnh để trích xuất các thực thể tài chính như số tiền, mục đích chi tiêu và tự động quy về hũ tài chính tương ứng.
    \item \textbf{Kỹ thuật gợi nhắc few-shot và suy luận từng bước:} Các mô hình ngôn ngữ lớn hiện đại có khả năng học nhanh từ một vài ví dụ minh họa được đưa vào lời nhắc (Few-shot Prompting) mà không cần huấn luyện lại \cite{brown2020-gpt3}, đồng thời có thể được dẫn dắt suy luận qua từng bước trung gian (Chain-of-Thought Prompting) để tăng độ chính xác cho các tác vụ đòi hỏi lập luận nhiều bước \cite{wei2022-cot}. Đây là hai kỹ thuật nền tảng cho cơ chế phân loại giao dịch bằng Few-shot Prompting được trình bày ở Chương 4.
    \item \textbf{Kiến trúc tác nhân suy luận và hành động (ReAct):} Thay vì chỉ trả lời trực tiếp từ kiến thức đã huấn luyện, nhiều hệ thống trợ lý ảo hiện đại kết hợp bước suy luận (reasoning) với bước gọi công cụ bên ngoài (acting) trong một vòng lặp lặp lại, được gọi là mô hình ReAct \cite{yao2022-react}. Cách tiếp cận này cho phép AI Agent tự quyết định khi nào cần truy vấn dữ liệu thực tế (ví dụ lịch sử chi tiêu) trước khi tổng hợp câu trả lời, là nền tảng cho kiến trúc tác nhân AI được trình bày ở Chương 3.
    \item \textbf{Hệ chuyên gia tư vấn:} AI Agent có thể gọi các công cụ truy vấn cơ sở dữ liệu (Tool-calling) để phân tích lịch sử chi tiêu, từ đó đưa ra các khuyến nghị tài chính cá nhân hóa như cắt giảm chi phí hưởng thụ hoặc tối ưu quỹ tiết kiệm.
\end{itemize}

\subsection{Trong hệ gợi ý học bổng (Scholarship Recommendation)}
Thay vì lọc dữ liệu theo điều kiện cứng, các nghiên cứu hiện đại áp dụng thuật toán so khớp thực thể và độ đo tương đồng văn bản:
\begin{itemize}
    \item \textbf{Fuzzy Matching và Tokenization:} Áp dụng cho các trường dữ liệu tên trường, tên ngành hoặc kỹ năng, giúp khắc phục các lỗi gõ sai, viết tắt hoặc khác biệt về cách dịch thuật ngữ (ví dụ: "IT", "CNTT", "Information Technology"). Chỉ số độ tương đồng Jaccard \cite{jaccard1901} và phương pháp N-gram \cite{cavnar1994-ngram} là hai độ đo nền tảng được ứng dụng rộng rãi trong tìm kiếm thông tin không chính xác. Bên cạnh đó, khoảng cách chỉnh sửa Levenshtein \cite{levenstein1966} là một độ đo kinh điển khác định lượng số phép biến đổi tối thiểu để chuyển một chuỗi ký tự thành chuỗi còn lại, là nền tảng lý thuyết cho các thuật toán so khớp chuỗi xấp xỉ như SequenceMatcher mà hệ thống sử dụng (xem Chương 4).
    \item \textbf{Kết hợp đa độ đo:} Các hệ thống so khớp hiện đại thường không dừng lại ở một độ đo duy nhất mà tổ hợp có trọng số giữa nhiều độ đo tương đồng (ví dụ Jaccard cấp độ token, Jaccard cấp độ bigram ký tự, tỷ lệ khớp chuỗi), nhằm cân bằng giữa độ nhạy cảm từ vựng cục bộ và khả năng chịu lỗi ở mức ký tự, giúp giảm thiểu sai sót với các thuật ngữ viết tắt phổ biến trong môi trường học thuật Việt Nam. Công thức trọng số cụ thể mà hệ thống Student360 áp dụng được trình bày chi tiết tại Chương 4.
    \item \textbf{Nhận diện ý định và phân loại (Intent Classification):} Sử dụng mô hình ngôn ngữ lớn Gemini \cite{gemini-api} thông qua thư viện LangChain \cite{langchain-docs} để xây dựng AI Agent phân loại ý định học bổng của người dùng. Kết quả phân loại ý định giúp Agent điều phối đúng các Tool truy vấn, nâng cao chất lượng câu trả lời tư vấn về học bổng trong hộp thoại AI.
\end{itemize}

\section{Các giải pháp thông báo đẩy thời gian thực trên nền tảng di động}
Hệ thống thông báo đẩy (Push Notification) đóng vai trò quan trọng trong việc tăng tỷ lệ giữ chân người dùng (Retention Rate) và đảm bảo thông tin khẩn cấp được truyền tải kịp thời. Các công nghệ phổ biến bao gồm:
\begin{itemize}
    \item \textbf{Firebase Cloud Messaging (FCM) \cite{fcm-docs}:} Dịch vụ đám mây trung gian của Google, hỗ trợ gửi thông báo đẩy đến các thiết bị di động (Android và iOS) ngay cả khi ứng dụng đã bị đóng (Background/Killed state). FCM tối ưu hóa thời lượng pin của thiết bị bằng cách gom nhóm các kết nối mạng ngầm của hệ điều hành. Đây là lựa chọn chính được hệ thống sử dụng để gửi thông báo học bổng và cảnh báo tài chính.
    \item \textbf{BullMQ và Redis Queue \cite{bullmq-docs}:} Để xử lý gửi thông báo đến hàng nghìn thiết bị mà không làm nghẽn tiến trình chính của máy chủ, các hệ thống áp dụng kiến trúc hàng đợi thông điệp (Message Queue). BullMQ kết hợp với Redis \cite{redis-docs} giúp quản lý thứ tự gửi, xử lý lỗi tự động (Automatic retries) và phân bổ tài nguyên gửi tin một cách tối ưu. Điểm khác biệt cốt lõi của hệ thống so với các ứng dụng tài chính thụ động là việc triển khai một kiến trúc thông báo \textbf{chủ động đa lớp} với nhiều loại thông báo Cron tự động chạy ngầm:
    \begin{enumerate}
        \item \textbf{Cảnh báo ngưỡng số dư hũ} (theo số tiền tuyệt đối hoặc phần trăm ngân sách): được kích hoạt ngay sau mỗi giao dịch chi tiêu theo kiểu fire-and-forget không chặn tiến trình API chính.
        \item \textbf{Cảnh báo ngân sách hũ sắp hết hoặc đã cạn}: gửi khi số dư ngân sách còn lại $\leq 10\%$ hoặc đã âm, được kích hoạt ngay sau giao dịch chi tiêu.
        \item \textbf{Nhắc lịch chuyển tiền tự động sắp đến}: gửi 24 giờ trước ngày thực hiện lịch để người dùng chuẩn bị trước, kèm xác nhận sau khi chuyển thành công hoặc thất bại.
        \item \textbf{Tóm tắt chi tiêu tuần} (Cron thứ Hai 8:00): gửi cho tất cả người dùng có giao dịch trong tuần, tổng hợp tổng thu, tổng chi và hũ chi nhiều nhất trong tuần.
        \item \textbf{Báo cáo tài chính cuối tháng} (Cron ngày 1 hàng tháng 8:00): gửi cho người dùng có hoạt động giao dịch trong tháng trước, nhắc xem phân tích AI 6 lọ.
        \item \textbf{Nhắc nhở cập nhật tài chính} (Cron thứ Hai 9:00): gửi cho những người dùng đã từng ghi nhận giao dịch nhưng không nhập bất kỳ giao dịch nào trong 7 ngày gần nhất.
        \item \textbf{Nhắc lịch chuyển tiền sắp kết thúc} (Cron thứ Hai 9:00): gửi cho các lịch chuyển tiền tự động sắp hết hạn trong vòng 3 ngày tới, nhắc người dùng gia hạn nếu cần.
    \end{enumerate}
    Chính nhờ kiến trúc này, hệ thống có khả năng chủ động tương tác với người dùng theo đúng ngữ cảnh sử dụng, thay vì chỉ phản ứng khi người dùng mở ứng dụng.
\end{itemize}
