\chapter{Phân tích và đặc tả chức năng nghiệp vụ}
\label{ch:dac-ta-ca-su-dung}
Chương này tập trung phân tích tổng quan các chức năng hệ thống, xác định các tác nhân tham gia tương tác, và đặc tả chi tiết các luồng chức năng nghiệp vụ cốt lõi của hệ thống hỗ trợ sinh viên Student360.

\section{Tổng quan chức năng hệ thống}
Như đã giới thiệu tại Chương 1, hệ thống Student360 gồm 4 phân hệ chức năng cốt lõi: Quản lý tài chính cá nhân, Trợ lý tài chính ảo AI, Quản lý học bổng trực tuyến và Quản lý học thuật/Bảng điểm. Các mục tiếp theo của chương này xác định các tác nhân tham gia, sơ đồ phân rã chức năng và đặc tả chi tiết nghiệp vụ của từng phân hệ.

\section{Các tác nhân tham gia hệ thống}
Hệ thống Student360 xác định 3 nhóm tác nhân (Actors) chính tham gia tương tác:
\begin{itemize}
    \item \textbf{Sinh viên (Student):} Tác nhân sử dụng phân hệ ứng dụng di động (Mobile App) để ghi chép chi tiêu, tương tác với AI Assistant, đăng ký và theo dõi học bổng, lập bảng điểm cá nhân.
    \item \textbf{Nhà tài trợ (Sponsor):} Tác nhân sử dụng phân hệ Webadmin để đăng tải và thiết lập tiêu chí học bổng, quản lý và duyệt hồ sơ ứng tuyển của sinh viên.
    \item \textbf{Quản trị viên (Admin):} Tác nhân sử dụng Webadmin để cấu hình danh mục trường, khoa, môn học (Master Data) và giám sát hoạt động hệ thống.
\end{itemize}

\section{Sơ đồ phân rã chức năng hệ thống}
Dưới đây là cấu trúc phân rã chức năng (Functional Decomposition) của hệ thống Student360 theo dạng hình cây phân cấp:
\begin{itemize}
    \item \textbf{Student360 System}
    \begin{itemize}
        \item \textbf{1. Phân hệ Quản lý tài chính (Mobile App)}
        \begin{itemize}
            \item 1.1 Cấu hình tỷ lệ 6 lọ tài chính
            \item 1.2 Ghi nhận giao dịch thủ công (Thu, Chi, Chuyển khoản)
            \item 1.3 Phân bổ thu nhập tự động
            \item 1.4 Thiết lập ngưỡng cảnh báo chi tiêu theo hũ
            \item 1.5 Lên lịch giao dịch định kỳ
        \end{itemize}
        \item \textbf{2. Phân hệ Trợ lý ảo AI (Mobile App)}
        \begin{itemize}
            \item 2.1 Trò chuyện thời gian thực qua stream SSE
            \item 2.2 Phân loại giao dịch tự động bằng LLM Classifier
            \item 2.3 Nhận cảnh báo chi tiêu bất thường
        \end{itemize}
        \item \textbf{3. Phân hệ Quản lý học bổng (Web \& Mobile)}
        \begin{itemize}
            \item 3.1 Quản lý chương trình học bổng (Nhà tài trợ)
            \item 3.2 Ứng tuyển học bổng trực tuyến (Sinh viên)
            \item 3.3 Cơ chế gợi ý học bổng tự động (AI Engine)
            \item 3.4 Xét duyệt hồ sơ ứng tuyển (Nhà tài trợ)
        \end{itemize}
        \item \textbf{4. Phân hệ Học thuật và bảng điểm (Web \& Mobile)}
        \begin{itemize}
            \item 4.1 Quản trị Master Data trường, khoa, môn học (Admin)
            \item 4.2 Lập mục tiêu tín chỉ học tập (Sinh viên)
            \item 4.3 Quản lý bảng điểm cá nhân (Sinh viên)
            \item 4.4 Nhập bảng điểm hàng loạt từ file CSV hoặc TXT (Sinh viên)
        \end{itemize}
    \end{itemize}
\end{itemize}

\section{Đặc tả chức năng nghiệp vụ}
Dưới đây là phần đặc tả chi tiết các chức năng nghiệp vụ cốt lõi của hệ thống Student360. Để tiện truy vết giữa phần đặc tả này với phần kiểm thử và đánh giá ở Chương~\ref{ch:hien-thuc-kiem-thu-danh-gia}, mỗi chức năng được gán một mã use case theo quy ước: tiền tố phân hệ (FIN -- Tài chính, AI -- Trợ lý AI, SCH -- Học bổng, ACA -- Học thuật) kết hợp số thứ tự, ví dụ FIN-UC01 hay SCH-UC03. Bảng truy vết tổng hợp được trình bày ở mục cuối chương.

\subsection{Quản lý tài chính cá nhân}
\noindent \textbf{Chức năng 1 (FIN-UC01): Cấu hình tỷ lệ 6 lọ tài chính}
\begin{itemize}
    \item \textbf{Tên chức năng:} Cấu hình tỷ lệ phân bổ 6 lọ tài chính.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên tùy chỉnh tỷ lệ phần trăm phân bổ giữa các hũ tài chính (6 hũ mặc định theo mô hình 6 Jars, hoặc hũ tùy chỉnh do sinh viên tự thêm), làm cơ sở để hệ thống tự động chia khoản thu nhập mới ở Chức năng "Phân bổ thu nhập tự động" bên dưới.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập thành công vào Mobile App.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình cấu hình hũ tài chính.
        \item Hệ thống hiển thị danh sách hũ hiện có kèm tỷ lệ phần trăm hiện tại và tổng tỷ lệ đang phân bổ.
        \item Sinh viên điều chỉnh tỷ lệ phần trăm cho từng hũ, có thể thêm hũ tùy chỉnh mới (đặt tên, biểu tượng, màu sắc) hoặc gỡ bỏ hũ không còn sử dụng.
        \item Sinh viên nhấn nút "Lưu".
        \item Mobile App gửi toàn bộ danh sách phân bổ vừa chỉnh sửa lên Backend để lưu lại, mỗi phần tử gồm mã hũ hiện có (nếu là hũ đã tồn tại), tên, tỷ lệ phần trăm, biểu tượng và màu sắc.
        \item Backend kiểm tra tổng tỷ lệ phần trăm của toàn bộ danh sách phải bằng đúng 100\% (sai lệch cho phép không quá 0.01), sau đó cập nhật hoặc tạo mới từng hũ tương ứng trong bảng \texttt{money\_jars}.
        \item Backend phản hồi danh sách hũ đã được cập nhật.
        \item Mobile App hiển thị lại danh sách hũ với tỷ lệ phân bổ mới.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Tổng tỷ lệ khác 100\%:} Backend từ chối toàn bộ yêu cầu và trả về lỗi 400 Bad Request. Mobile App hiển thị thông báo: "Tổng tỷ lệ phân bổ phải bằng 100\%".
        \item \textit{Ngoại lệ 2 - Cập nhật hũ không thuộc sở hữu:} Nếu mã hũ gửi lên không thuộc về sinh viên hiện tại, Backend trả về lỗi 404 Not Found.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Tỷ lệ phân bổ mới được lưu vào cơ sở dữ liệu và áp dụng ngay cho các lần phân bổ thu nhập tự động tiếp theo.
\end{itemize}

\noindent \textbf{Chức năng 2 (FIN-UC02): Ghi nhận giao dịch thủ công}
\begin{itemize}
    \item \textbf{Tên chức năng:} Ghi nhận giao dịch tài chính thủ công (Thu, Chi, Chuyển khoản).
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên tự nhập trực tiếp một giao dịch Thu nhập, Chi tiêu hoặc Chuyển khoản giữa các hũ mà không cần qua bước phân tích của AI -- khác với Chức năng "Phân loại giao dịch bằng AI" ở mục Trợ lý AI tài chính, vốn tự động trích xuất thông tin từ một câu mô tả bằng ngôn ngữ tự nhiên.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập và có ít nhất một hũ tài chính.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên chọn loại giao dịch (Thu nhập, Chi tiêu hoặc Chuyển khoản) trên màn hình tương ứng.
        \item Sinh viên nhập số tiền, chọn hũ nguồn; riêng với Chuyển khoản, sinh viên chọn thêm hũ đích. Sinh viên có thể chọn nhãn chi tiêu (mặc định là nhãn "Khác" nếu bỏ trống), nhập ngày giao dịch và mô tả/ghi chú.
        \item Sinh viên nhấn nút "Lưu giao dịch".
        \item Mobile App gửi yêu cầu tạo giao dịch mới kèm toàn bộ dữ liệu đã nhập lên Backend.
        \item Backend xác thực dữ liệu (số tiền phải lớn hơn 0, các hũ tham chiếu phải thuộc về sinh viên hiện tại), tạo bản ghi mới trong bảng \texttt{financial\_transactions}.
        \item Backend phản hồi thông tin giao dịch vừa tạo về Mobile App.
        \item Mobile App cập nhật danh sách lịch sử giao dịch và số dư hũ liên quan theo cơ chế tính động đã mô tả ở Chức năng "Phân bổ thu nhập tự động" bên dưới.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Số tiền không hợp lệ:} Sinh viên nhập số tiền bằng 0 hoặc số âm, Backend từ chối yêu cầu và trả về lỗi 400 Bad Request.
        \item \textit{Ngoại lệ 2 - Hũ không thuộc sở hữu:} Hũ nguồn hoặc hũ đích không thuộc về sinh viên hiện tại, Backend trả về lỗi 404 Not Found.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Giao dịch mới được lưu vào cơ sở dữ liệu và số dư các hũ liên quan được cập nhật ngay lập tức. Thao tác sửa và xóa giao dịch (vuốt trái trong màn hình lịch sử) được đánh giá riêng tại Chương 6.
\end{itemize}

\noindent \textbf{Chức năng 3 (FIN-UC03): Phân bổ thu nhập tự động}
\begin{itemize}
    \item \textbf{Tên chức năng:} Phân bổ thu nhập tự động theo mô hình 6 lọ.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên ghi nhận một khoản thu mới và tự động tính toán, phân chia số tiền đó vào 6 hũ tài chính theo tỷ lệ phần trăm đã được cấu hình từ trước trên hệ thống.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập thành công vào Mobile App và đã thiết lập cấu hình tỷ lệ phần trăm cho 6 hũ tài chính (đảm bảo tổng tỷ lệ bằng 100\%).
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên nhấn nút "Thêm thu nhập" tại màn hình Tổng quan ví.
        \item Hệ thống hiển thị biểu mẫu nhập thông tin gồm: Số tiền thu nhập, nguồn thu nhập, mô tả chi tiết, và tùy chọn phân bổ.
        \item Sinh viên nhập số tiền (ví dụ: $10.000.000$ VNĐ), chọn danh mục nguồn thu (ví dụ: "Lương làm thêm"), viết mô tả và bật tùy chọn "Tự động phân bổ".
        \item Sinh viên nhấn nút "Xác nhận".
        \item Mobile App gửi yêu cầu phân bổ thu nhập kèm thông tin số tiền và mô tả lên Backend.
        \item Backend nhận yêu cầu, truy vấn bảng \texttt{money\_jars} để lấy cấu hình tỷ lệ phần trăm của sinh viên hiện tại (ví dụ: 55\%, 10\%, 10\%, 10\%, 10\%, 5\%).
        \item Backend tính toán số tiền phân bổ cụ thể cho từng hũ tài chính, đồng thời kiểm tra tổng số tiền đã phân bổ cho các hũ phải khớp chính xác với số tiền thu nhập ban đầu.
        \item Backend thực hiện đồng loạt 6 lệnh ghi bản ghi giao dịch thu vào cơ sở dữ liệu, mỗi lệnh cập nhật số dư của một hũ tương ứng.
        \item Sau khi toàn bộ 6 lệnh ghi hoàn thành, Backend phản hồi kết quả về Mobile App.
        \item Mobile App cập nhật trạng thái dữ liệu (qua React Query) và hiển thị thông báo phân bổ tiền thành công kèm chi tiết số tiền từng lọ nhận được.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Số tiền không hợp lệ:} Tại bước 3, sinh viên nhập số tiền bằng 0 hoặc số âm. Hệ thống hiển thị lỗi ngay tại giao diện di động: "Số tiền thu nhập phải lớn hơn 0" và không cho phép gửi yêu cầu.
        \item \textit{Ngoại lệ 2 - Tổng số tiền phân bổ không khớp:} Tại bước 6-7, nếu client tự gửi kèm danh sách phân bổ chi tiết theo từng hũ (\texttt{jar\allowbreak Allocations}) mà tổng số tiền của danh sách này không bằng đúng số tiền thu nhập ban đầu, Backend hủy tiến trình, trả về lỗi. Mobile App hiển thị thông báo: "Số tiền phân bổ không khớp với số tiền thu nhập, vui lòng kiểm tra lại".
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Số dư khả dụng của từng hũ được cập nhật tăng tương ứng. 6 bản ghi giao dịch thu nhập mới được lưu vào cơ sở dữ liệu PostgreSQL. Hệ thống đồng bộ dữ liệu số dư mới lên màn hình Tổng quan ví.
    \item \textbf{Cơ chế đảm bảo tính toàn vẹn số dư hũ:} Số dư hiện tại của mỗi hũ được lưu tại trường \texttt{current\_balance} để phục vụ truy vấn và hiển thị. Sau thao tác làm thay đổi giao dịch, Backend tính lại giá trị từ lịch sử giao dịch thực tế và cập nhật trường này theo công thức tích lũy:
    $$\text{Số dư hũ} = \sum (\text{Thu vào}) - \sum (\text{Chi ra}) \pm \sum (\text{Chuyển quỹ})$$
    Cơ chế tính lại giúp khôi phục tính nhất quán nếu giá trị số dư lưu sẵn bị sai lệch. Một trường hợp đặc biệt liên quan là khi sinh viên xóa một hũ tài chính tùy chỉnh đang có số dư khác 0: hệ thống bắt buộc chọn một hũ đích để nhận lại tiền, sau đó tạo đồng thời một cặp giao dịch bù trừ (trừ tiền ở hũ nguồn dưới dạng giao dịch \texttt{expense}, cộng tiền vào hũ đích dưới dạng giao dịch \texttt{income}) để tổng tài sản của người dùng không đổi, rồi tính lại số dư của cả hai hũ liên quan.
\end{itemize}

\noindent \textbf{Chức năng 4 (FIN-UC04): Thiết lập ngưỡng cảnh báo chi tiêu theo hũ}
\begin{itemize}
    \item \textbf{Tên chức năng:} Thiết lập ngưỡng cảnh báo chi tiêu cho từng hũ tài chính.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên cấu hình một ngưỡng cảnh báo cho từng hũ tài chính theo một trong hai chế độ độc lập: (a) cảnh báo khi số dư hũ giảm xuống dưới một số tiền cố định, hoặc (b) cảnh báo khi tỷ lệ đã chi vượt quá một phần trăm nhất định. Khác với mô hình "hạn mức ngân sách theo tháng" ban đầu, cấu hình này không giới hạn theo một tháng/năm cụ thể mà áp dụng liên tục cho tới khi sinh viên thay đổi lại.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập thành công vào Mobile App và đã có ít nhất một hũ tài chính.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập vào màn hình "Chi tiết hũ" của một hũ bất kỳ (ví dụ: hũ Hưởng thụ - \texttt{enjoyment}).
        \item Sinh viên nhấn chọn "Cài đặt cảnh báo chi tiêu".
        \item Hệ thống hiển thị giao diện với hai lựa chọn độc lập có thể bật/tắt riêng: "Cảnh báo theo số dư tối thiểu" (nhập số tiền, ví dụ $200.000$ VNĐ) và "Cảnh báo theo phần trăm đã chi" (nhập số nguyên từ 0-100, ví dụ 80\%).
        \item Sinh viên bật một hoặc cả hai lựa chọn, nhập giá trị tương ứng và nhấn "Lưu".
        \item Mobile App gửi yêu cầu lưu cấu hình cảnh báo vừa thiết lập lên Backend.
        \item Backend xác thực giá trị đầu vào (phần trăm phải là số nguyên trong khoảng 0-100, số tiền phải không âm), sau đó tạo mới hoặc cập nhật bản ghi duy nhất tương ứng với hũ trong bảng \texttt{jar\_\allowbreak notification\_\allowbreak settings}.
        \item Backend phản hồi thành công về Mobile App.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Giá trị nhập không hợp lệ:} Nếu sinh viên nhập phần trăm ngoài khoảng 0-100 hoặc số tiền âm, Backend từ chối yêu cầu ở tầng validate DTO và trả về lỗi 400 Bad Request. Mobile App hiển thị thông báo lỗi tương ứng ngay tại form nhập liệu.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Bản ghi ngưỡng cảnh báo được cập nhật hoặc tạo mới trong bảng \texttt{jar\_notification\_settings}. Từ giao dịch chi tiêu tiếp theo trên hũ đó, hệ thống sẽ kiểm tra ngưỡng này ngay trong luồng tạo giao dịch (đồng bộ, không cần chờ job quét định kỳ) và gửi cảnh báo qua hàng đợi thông báo đẩy nếu vượt ngưỡng.
    \item \textbf{Cơ chế Idempotency cho kênh cảnh báo bất thường tự động:} Bên cạnh ngưỡng cảnh báo do sinh viên tự cấu hình ở trên, hệ thống còn một kênh cảnh báo riêng do Backend tự động phát hiện: một Cron Job chạy định kỳ lúc 23:00 hằng ngày quét các giao dịch bất thường trong ngày. Vì job này có thể chạy lại hoặc xử lý trùng, hệ thống áp dụng cơ chế loại bỏ trùng lặp (Deduplication) dựa trên khóa Idempotency tính từ tổ hợp \texttt{user\_id + target\_id} (\texttt{target\_id} là mã giao dịch đang được đánh giá): trước khi thêm một cảnh báo mới vào bảng \texttt{ai\_anomaly\_alerts} và kích hoạt gửi thông báo đẩy, hệ thống kiểm tra đã tồn tại cảnh báo nào cùng khóa này chưa, nếu có thì bỏ qua không tạo mới.
\end{itemize}

\noindent \textbf{Thuật toán tính điểm số so khớp tương thích}

Điểm số so khớp tương thích giữa hồ sơ sinh viên và một chương trình học bổng (nhắc tới ở Chức năng 2 bên dưới) được AI Service tính trực tiếp trên bộ nhớ RAM tại thời điểm truy vấn, không phụ thuộc cơ sở dữ liệu vector. Hệ thống gộp các trường thông tin có giá trị trong hồ sơ sinh viên (ngành học, trường, kỹ năng, hoạt động ngoại khóa) thành một đoạn văn bản, chuẩn hóa Unicode để đồng nhất cách hiển thị tiếng Việt trên các hệ điều hành khác nhau, rồi so sánh với tiêu đề và mô tả của học bổng bằng ba độ đo tương đồng văn bản kinh điển:
\begin{itemize}
    \item \textbf{Độ tương đồng Jaccard theo token} \cite{jaccard1901} (trọng số 0.5): so sánh tập từ vựng xuất hiện trong hồ sơ và trong học bổng, điểm càng cao khi hai văn bản dùng chung nhiều từ khóa.
    \item \textbf{Độ tương đồng Jaccard theo Bigram ký tự} \cite{cavnar1994-ngram} (trọng số 0.3): so sánh theo từng cặp ký tự liền kề thay vì theo từ, giúp nhận diện các trường hợp viết tắt hoặc sai chính tả mà so khớp theo từ không phát hiện được.
    \item \textbf{Tỷ lệ khớp chuỗi} bằng mô-đun \texttt{difflib.SequenceMatcher} của Python (trọng số 0.2): đo mức độ giống nhau tổng thể giữa hai chuỗi văn bản.
\end{itemize}
Ba độ đo trên được kết hợp theo trọng số cố định (0.5/0.3/0.2) thành một điểm số tổng hợp duy nhất trong khoảng từ 0 đến 1, hiển thị cho sinh viên dưới dạng phần trăm khớp (ví dụ: Khớp 85\%) ở Chức năng 2 bên dưới. Nếu hồ sơ sinh viên hoặc mô tả học bổng không có đủ dữ liệu văn bản để so sánh, hệ thống trả về điểm khớp bằng 0.

\noindent \textbf{Chức năng 2 (SCH-UC02): Ứng tuyển học bổng trực tuyến}
\begin{itemize}
    \item \textbf{Tên chức năng:} Sinh viên ứng tuyển học bổng trực tuyến.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên duyệt danh sách học bổng, xem mức độ tương thích hồ sơ do AI gợi ý, điền thông tin đăng ký và tải lên các tài liệu minh chứng để gửi hồ sơ ứng tuyển tới nhà tài trợ.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã tạo hồ sơ học thuật cá nhân. Chương trình học bổng vẫn còn trong thời gian nhận hồ sơ.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Học bổng" trên ứng dụng di động.
        \item Hệ thống hiển thị danh sách các học bổng đang mở kèm điểm số so khớp tương thích của AI (ví dụ: Khớp 85\%).
        \item Sinh viên nhấn chọn một học bổng cụ thể để xem chi tiết tiêu chí và danh sách các tài liệu bắt buộc phải nộp.
        \item Sinh viên nhấn nút "Ứng tuyển ngay".
        \item Hệ thống hiển thị biểu mẫu đăng ký bao gồm các thông tin cá nhân lấy từ profile và các nút upload file tương ứng với từng tài liệu yêu cầu.
        \item Sinh viên điền thông tin bổ sung, tải lên các file tài liệu dạng PDF/Image từ điện thoại. Việc upload file thực tế do Mobile App tự thực hiện trực tiếp lên dịch vụ lưu trữ file trước, sau đó chỉ gửi đường dẫn (\texttt{fileUrl}) đã có sẵn cho Backend -- Backend không đóng vai trò trung chuyển nội dung file mà chỉ nhận và lưu URL.
        \item Sinh viên nhấn nút "Nộp hồ sơ".
        \item Mobile App gửi request lên Backend NestJS chứa thông tin đăng ký và danh sách URL file đã upload trước đó.
        \item Backend xác thực tính hợp lệ của thông tin đăng ký và định dạng các URL tài liệu được cung cấp.
        \item Backend lưu thông tin hồ sơ ứng tuyển vào bảng \texttt{student\_\allowbreak scholarships}, lưu các URL file đính kèm vào bảng \texttt{student\_\allowbreak scholarship\_\allowbreak documents}, và lưu lịch sử chuyển trạng thái sang \texttt{submitted} vào bảng \texttt{student\_\allowbreak scholarship\_\allowbreak history}.
        \item Mobile App chuyển giao diện về trạng thái "Đã nộp" và hiển thị thông báo ứng tuyển thành công.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Luồng phụ - Lưu nháp hồ sơ:} Tại bước 7, sinh viên chưa chuẩn bị đủ tài liệu có thể nhấn nút "Lưu nháp". Hệ thống sẽ lưu hồ sơ với trạng thái "Bản nháp" (\texttt{draft}) để sinh viên có thể tiếp tục chỉnh sửa và nộp sau.
        \item \textit{Ngoại lệ 1 - Nộp hồ sơ trùng lặp:} Tại bước 9, Backend kiểm tra và phát hiện sinh viên đã có một hồ sơ ứng tuyển cho chương trình học bổng này ở trạng thái không còn cho phép chỉnh sửa (đã \texttt{submitted}, \texttt{under\_review}, \texttt{approved}, \texttt{rejected} hoặc \texttt{awarded}). Backend trả về lỗi HTTP 409 Conflict. Mobile App hiển thị thông báo: "Bạn đã nộp hồ sơ cho học bổng này từ trước, không thể nộp lại". (Nếu hồ sơ cũ đang ở trạng thái \texttt{draft}, \texttt{need\_modification} hoặc \texttt{cancelled}, Backend sẽ cập nhật ngay trên hồ sơ đó thay vì báo lỗi trùng lặp.)
        \item \textit{Ngoại lệ 2 - Dung lượng file vượt quá giới hạn:} Do việc upload file diễn ra trực tiếp từ Mobile App lên dịch vụ lưu trữ (không qua Backend), việc kiểm tra dung lượng file vượt giới hạn được thực hiện ngay tại bước tải lên trên thiết bị di động trước khi gửi yêu cầu nộp hồ sơ. Nếu vượt giới hạn, ứng dụng hiển thị thông báo: "Tài liệu vượt quá dung lượng cho phép. Vui lòng nén file và thử lại" và không cho phép tiếp tục.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Một bản ghi hồ sơ ứng tuyển mới được tạo ở trạng thái "Đã nộp" (\texttt{submitted}) và được lưu thành công trên hệ thống.
\end{itemize}

\noindent \textbf{Chức năng 3 (SCH-UC03): Xét duyệt hồ sơ học bổng}
\begin{itemize}
    \item \textbf{Tên chức năng:} Nhà tài trợ xét duyệt hồ sơ ứng tuyển học bổng.
    \item \textbf{Tác nhân (Actor):} Nhà tài trợ (Sponsor).
    \item \textbf{Mô tả:} Nhà tài trợ xem danh sách các hồ sơ ứng tuyển của sinh viên gửi tới chương trình học bổng của tổ chức mình, xem chi tiết thông tin học thuật, tải các file tài liệu đính kèm để đối chiếu và thay đổi trạng thái hồ sơ.
    \item \textbf{Điều kiện trước (Preconditions):} Nhà tài trợ đã được phân quyền quản trị của tổ chức tài trợ và đã đăng nhập thành công vào Webadmin Portal.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Nhà tài trợ truy cập vào màn hình "Quản lý hồ sơ ứng tuyển" trên Webadmin.
        \item Hệ thống hiển thị danh sách các hồ sơ đang ở trạng thái chờ duyệt.
        \item Nhà tài trợ nhấn chọn một hồ sơ cụ thể để xem chi tiết.
        \item Hệ thống hiển thị toàn bộ thông tin học thuật của sinh viên kèm danh sách các file tài liệu minh chứng sinh viên đã nộp. Nhà tài trợ click mở các file đính kèm để đối soát.
        \item Nhà tài trợ đưa ra quyết định xét duyệt bằng cách chọn một trong ba tùy chọn:
        \begin{itemize}
            \item \textbf{Chấp nhận cấp học bổng:} Cập nhật trạng thái sang "Chấp nhận" (\texttt{approved}).
            \item \textbf{Yêu cầu chỉnh sửa:} Cập nhật trạng thái sang "Cần chỉnh sửa" (\texttt{need\_modification}) kèm nội dung ghi chú yêu cầu.
            \item \textbf{Từ chối:} Cập nhật trạng thái sang "Từ chối" (\texttt{rejected}) kèm lý do từ chối.
        \end{itemize}
        \item Nhà tài trợ nhấn nút "Cập nhật".
        \item Webadmin gửi yêu cầu cập nhật trạng thái hồ sơ kèm trạng thái mới và ghi chú duyệt lên Backend.
        \item Backend thực thi một Transaction cập nhật trạng thái hồ sơ, ghi nhận lịch sử vào bảng \texttt{student\_scholarship\_history}.
        \item Backend tạo một tiến trình Job gửi thông báo đẩy và đẩy vào hàng đợi BullMQ.
        \item Worker ngầm của hệ thống lấy Job ra, gọi FCM SDK để gửi thông báo tới thiết bị di động của sinh viên tương ứng.
        \item Webadmin cập nhật danh sách hồ sơ và hiển thị thông báo "Cập nhật trạng thái hồ sơ thành công".
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Quyền truy cập không hợp lệ:} Tại bước 7, Backend kiểm tra thấy nhà tài trợ của tổ chức A đang cố gắng cập nhật trạng thái hồ sơ nộp vào chương trình học bổng của tổ chức B. Backend từ chối yêu cầu, trả về lỗi HTTP 403 Forbidden. Webadmin hiển thị thông báo: "Bạn không có quyền thực hiện thao tác này".
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Trạng thái hồ sơ được cập nhật và sinh viên nhận được thông báo kết quả.
\end{itemize}

\noindent \textbf{Vòng đời trạng thái của hồ sơ ứng tuyển}

Ba chức năng ở trên (ứng tuyển, lưu nháp và xét duyệt) cùng thao tác trên một đối tượng hồ sơ ứng tuyển, nên trạng thái của hồ sơ được quản lý theo một mô hình trạng thái (state machine) thống nhất thay vì rải rác ở từng luồng. Bảng~\ref{tab:scholarship-state} tổng hợp lại toàn bộ các bước chuyển trạng thái hợp lệ, kèm tác nhân được phép thực hiện, điều kiện ràng buộc (guard) và hệ quả đi kèm. Mọi bước chuyển không nằm trong bảng này đều bị hệ thống từ chối.

\subsection{Quản lý dữ liệu học thuật}
\noindent \textbf{Chức năng 1 (ACA-UC01): Quản trị viên quản lý Master Data trường, khoa/chương trình đào tạo và môn học}
\begin{itemize}
    \item \textbf{Tên chức năng:} Quản trị danh mục dữ liệu học thuật chuẩn hóa (Master Data).
    \item \textbf{Tác nhân (Actor):} Quản trị viên.
    \item \textbf{Mô tả:} Cho phép quản trị viên tạo, cập nhật, xóa và import hàng loạt danh mục trường đại học, khoa/chương trình đào tạo và môn học trên Webadmin. Danh mục này là nguồn dữ liệu chuẩn hóa duy nhất được cả Webadmin và Mobile App sử dụng để phục vụ đăng ký thông tin học thuật, nhập điểm và đối chiếu điều kiện xét duyệt học bổng.
    \item \textbf{Điều kiện trước (Preconditions):} Quản trị viên đã đăng nhập thành công vào Webadmin Portal với quyền quản trị hệ thống.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Quản trị viên quản lý danh mục trường: Webadmin cho phép xem danh sách, tạo mới, cập nhật và xóa từng trường đại học.
        \item Quản trị viên quản lý khoa/chương trình đào tạo: với từng trường, Webadmin cho phép xem danh sách, thêm mới, cập nhật, xóa từng khoa/chương trình đào tạo, hoặc import hàng loạt từ file.
        \item Quản trị viên quản lý môn học: với từng trường, Webadmin cho phép xem danh sách, thêm mới, cập nhật, xóa từng môn học, hoặc import hàng loạt từ file.
        \item Dữ liệu vừa cập nhật lập tức khả dụng cho Mobile App: sinh viên có thể chọn trường, khoa/chương trình đào tạo và môn học từ danh mục mới nhất khi khai báo hồ sơ cá nhân hoặc nhập điểm.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Import hàng loạt có bản ghi lỗi:} Khi import file dữ liệu hàng loạt, nếu một số dòng có dữ liệu không hợp lệ (thiếu trường bắt buộc, trùng mã), Backend trả về danh sách các dòng lỗi kèm lý do cụ thể thay vì từ chối toàn bộ file, cho phép quản trị viên sửa và import lại riêng các dòng lỗi.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Danh mục Master Data được cập nhật và đồng bộ ngay cho cả Webadmin lẫn Mobile App.
\end{itemize}

\noindent \textbf{Chức năng 2 (ACA-UC02): Quản lý bảng điểm cá nhân}
\begin{itemize}
    \item \textbf{Tên chức năng:} Quản lý bảng điểm cá nhân và tự động tính toán GPA.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên tự cập nhật điểm số các môn học đã hoàn thành của mình trên ứng dụng di động. Hệ thống sẽ tự động tính toán điểm trung bình tích lũy GPA dựa trên số tín chỉ của từng môn học được tham chiếu từ danh mục môn học chuẩn hóa của trường.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập và đã cấu hình thông tin trường đại học trong hồ sơ cá nhân.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Góc học tập" và chọn tab "Bảng điểm".
        \item Hệ thống hiển thị danh sách các môn học đã nhập điểm kèm số tín chỉ và điểm số cụ thể.
        \item Sinh viên nhấn nút "Thêm điểm môn học".
        \item Hệ thống hiển thị biểu mẫu bao gồm: Danh sách môn học chuẩn, Ô nhập điểm số.
        \item Sinh viên gõ tìm kiếm và chọn môn học (ví dụ: "Nhập môn Lập trình - 4 tín chỉ"), nhập điểm số đạt được (ví dụ: 8.5 hoặc điểm chữ A) và nhấn "Lưu".
        \item Mobile App gửi thông tin môn học và điểm số vừa nhập lên Backend để lưu lại.
        \item Backend xác thực môn học thuộc về trường đại học sinh viên đang học, kiểm tra điểm số có nằm trong giới hạn thang điểm của môn học không.
        \item Backend lưu bản ghi điểm môn học và phản hồi thành công về Mobile App. Điểm GPA tích lũy \textbf{không} được ghi đè trực tiếp vào hồ sơ học thuật của sinh viên ngay tại bước này.
        \item Khi sinh viên mở màn hình Dashboard bảng điểm, Backend mới truy vấn toàn bộ các môn học có điểm của sinh viên, lấy số tín chỉ và điểm số của từng môn để tính GPA tích lũy có trọng số theo công thức sau, tính trực tiếp tại thời điểm truy vấn (không lưu lại giá trị này vào cơ sở dữ liệu):
        $$\text{GPA} = \frac{\sum (\text{Điểm môn } i \times \text{Số tín chỉ môn } i)}{\sum (\text{Số tín chỉ})}$$
        \item Mobile App hiển thị bảng điểm được cập nhật mới kèm chỉ số GPA tích lũy tính theo thời gian thực (ví dụ: GPA: 3.42 / 4.0).
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Điểm số không hợp lệ:} Tại bước 5, sinh viên nhập điểm là 11 (vượt quá thang điểm 10) hoặc nhập số âm. Hệ thống hiển thị thông báo lỗi ngay trên giao diện nhập: "Điểm số nhập vào không hợp lệ. Vui lòng nhập điểm trong khoảng từ 0 đến 10".
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Bảng điểm và GPA tích lũy được cập nhật tức thời trên màn hình Dashboard học tập.
\end{itemize}

\noindent \textbf{Chức năng 3 (ACA-UC03): Nhập điểm hàng loạt từ file}
\begin{itemize}
    \item \textbf{Tên chức năng:} Nhập điểm sinh viên hàng loạt từ file bảng điểm.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên tải lên một file văn bản (CSV hoặc TXT) chứa danh sách các môn học và điểm số kết xuất từ trang web đào tạo của trường để hệ thống tự động phân tích và lưu hàng loạt môn học vào bảng điểm cá nhân.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên có file bảng điểm đúng định dạng hỗ trợ. Thiết bị di động có quyền truy cập tệp tin.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Bảng điểm" và chọn chức năng "Nhập điểm từ tệp tin".
        \item Hệ thống hiển thị hộp thoại yêu cầu chọn tệp tin và hướng dẫn định dạng file mẫu.
        \item Sinh viên nhấn chọn file từ bộ nhớ thiết bị.
        \item Mobile App thực hiện đọc nội dung tệp tin cục bộ, sử dụng bộ Parser phân tích cú pháp từng dòng của tệp để trích xuất ra các cột thông tin: Mã môn học, Điểm số.
        \item Hệ thống hiển thị danh sách các môn học và điểm số đã parse được lên màn hình xem trước để sinh viên rà soát lại.
        \item Sinh viên kiểm tra thông tin và nhấn nút "Xác nhận lưu tất cả".
        \item Mobile App gửi toàn bộ danh sách các cặp giá trị (Mã môn học, Điểm số) đã được parse lên Backend trong một lần gửi duy nhất, tái sử dụng cùng luồng lưu điểm với Chức năng 2 ở trên chứ không qua một luồng nhập hàng loạt tách biệt riêng.
        \item Backend kiểm tra sự tồn tại của các mã môn học trong danh mục Master Data của trường đại học mà sinh viên đăng ký.
        \item Backend lặp qua từng phần tử trong danh sách để xóa điểm cũ của những môn học trùng lặp (nếu có) và thêm mới các bản ghi điểm môn học.
        \item Backend phản hồi kết quả thành công kèm số lượng môn học đã lưu. GPA tích lũy không được tính và lưu lại ở bước này, mà sẽ được tính theo thời gian thực khi sinh viên mở lại Dashboard bảng điểm, theo đúng cơ chế đã mô tả ở Chức năng 2.
        \item Mobile App hiển thị thông báo "Đã nhập thành công X môn học vào bảng điểm" và tải lại giao diện bảng điểm mới.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Định dạng tệp tin không được hỗ trợ:} Tại bước 4, Parser cục bộ phát hiện tệp tin tải lên có phần mở rộng không hợp lệ hoặc cấu trúc nội dung không thể đọc được. Hệ thống dừng tiến trình và hiển thị thông báo lỗi: "Tệp tin không đúng định dạng. Vui lòng tải lên file CSV hoặc TXT theo đúng cấu trúc hướng dẫn".
        \item \textit{Ngoại lệ 2 - Mã môn học không tồn tại trong Master Data:} Tại bước 8, Backend phát hiện có mã môn học trong danh sách gửi lên không tồn tại trong danh mục môn học chuẩn của trường đại học đó. Backend sẽ trả về danh sách các môn học bị lỗi. Mobile App hiển thị hộp thoại thông báo: "Phát hiện một số môn học có mã không hợp lệ. Vui lòng kiểm tra lại mã môn học hoặc liên hệ quản trị viên để cập nhật danh mục môn học".
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Danh sách môn học và điểm số trích xuất thành công được ghi nhận hàng loạt vào cơ sở dữ liệu, GPA tích lũy được cập nhật lại.
\end{itemize}

Cuối cùng, hai nhóm SCH-UC01 và ACA-UC01--03 hiện mới được kiểm thử gộp trong mục kiểm thử hộp đen chung (Mục~\ref{sec:kiem-thu-chuc-nang}) chứ chưa có mã ca kiểm thử riêng; việc bổ sung ca kiểm thử được mã hóa cho các nhóm này là một hạng mục hoàn thiện tiếp theo khi nhóm xây dựng bộ kiểm thử tự động đầy đủ hơn, như đã nêu trong hướng phát triển ở Chương 7.
