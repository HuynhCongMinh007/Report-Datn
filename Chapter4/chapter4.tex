\chapter{Phân tích và đặc tả chức năng nghiệp vụ}
\label{ch:dac-ta-ca-su-dung}
Chương này tập trung phân tích tổng quan các chức năng hệ thống, xác định các tác nhân tham gia tương tác, và đặc tả chi tiết các luồng chức năng nghiệp vụ cốt lõi của hệ thống hỗ trợ sinh viên Student360.

\section{Tổng quan chức năng hệ thống}
Như đã giới thiệu tại Chương 1, hệ thống Student360 gồm 4 phân hệ chức năng cốt lõi: Quản lý tài chính cá nhân, Trợ lý tài chính ảo AI, Quản lý học bổng trực tuyến và Quản lý học thuật/Bảng điểm. Các mục tiếp theo của chương này xác định các tác nhân tham gia, sơ đồ phân rã chức năng và đặc tả chi tiết nghiệp vụ của từng phân hệ.

\section{Các tác nhân tham gia hệ thống}
Hệ thống Student360 xác định 3 nhóm tác nhân (Actors) chính tham gia tương tác:
\begin{itemize}
    \item \textbf{Sinh viên (Student):} Tác nhân sử dụng phân hệ ứng dụng di động (Mobile App) để ghi chép chi tiêu, tương tác với AI Assistant, đăng ký và theo dõi học bổng, lập bảng điểm cá nhân.
    \item \textbf{Nhà tài trợ (Sponsor):} Tác nhân sử dụng phân hệ Webadmin để đăng tải và thiết lập tiêu chí học bổng, quản lý và duyệt hồ sơ ứng tuyển của sinh viên.
    \item \textbf{Quản trị viên (Admin):} Tác nhân sử dụng Webadmin để cấu hình danh mục trường, khoa, môn học (Master Data) và giám sát hoạt động hệ thống.
\end{itemize}

\section{Sơ đồ phân rã chức năng hệ thống}
Dưới đây là cấu trúc phân rã chức năng (Functional Decomposition) của hệ thống Student360 theo dạng hình cây phân cấp:
\begin{itemize}
    \item \textbf{Student360 System}
    \begin{itemize}
        \item \textbf{1. Phân hệ Quản lý tài chính (Mobile App)}
        \begin{itemize}
            \item 1.1 Cấu hình tỷ lệ 6 lọ tài chính
            \item 1.2 Ghi nhận giao dịch thủ công (Thu, Chi, Chuyển khoản)
            \item 1.3 Phân bổ thu nhập tự động
            \item 1.4 Thiết lập ngưỡng cảnh báo chi tiêu theo hũ
            \item 1.5 Lên lịch giao dịch định kỳ
        \end{itemize}
        \item \textbf{2. Phân hệ Trợ lý ảo AI (Mobile App)}
        \begin{itemize}
            \item 2.1 Trò chuyện thời gian thực qua stream SSE
            \item 2.2 Phân loại giao dịch tự động bằng LLM Classifier
            \item 2.3 Nhận cảnh báo chi tiêu bất thường
        \end{itemize}
        \item \textbf{3. Phân hệ Quản lý học bổng (Web \& Mobile)}
        \begin{itemize}
            \item 3.1 Quản lý chương trình học bổng (Nhà tài trợ)
            \item 3.2 Ứng tuyển học bổng trực tuyến (Sinh viên)
            \item 3.3 Cơ chế gợi ý học bổng tự động (AI Engine)
            \item 3.4 Xét duyệt hồ sơ ứng tuyển (Nhà tài trợ)
        \end{itemize}
        \item \textbf{4. Phân hệ Học thuật và bảng điểm (Web \& Mobile)}
        \begin{itemize}
            \item 4.1 Quản trị Master Data trường, khoa, môn học (Admin)
            \item 4.2 Lập mục tiêu tín chỉ học tập (Sinh viên)
            \item 4.3 Quản lý bảng điểm cá nhân (Sinh viên)
            \item 4.4 Nhập bảng điểm hàng loạt từ file CSV/TXT (Sinh viên)
        \end{itemize}
    \end{itemize}
\end{itemize}

\section{Đặc tả chức năng nghiệp vụ}
Dưới đây là phần đặc tả chi tiết các chức năng nghiệp vụ cốt lõi của hệ thống Student360.

\subsection{Quản lý tài chính cá nhân}
\noindent \textbf{Chức năng 1: Phân bổ thu nhập tự động}
\begin{itemize}
    \item \textbf{Tên chức năng:} Phân bổ thu nhập tự động theo mô hình 6 lọ.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên ghi nhận một khoản thu mới và tự động tính toán, phân chia số tiền đó vào 6 hũ tài chính theo tỷ lệ phần trăm đã được cấu hình từ trước trên hệ thống.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập thành công vào Mobile App và đã thiết lập cấu hình tỷ lệ phần trăm cho 6 hũ tài chính (đảm bảo tổng tỷ lệ bằng 100\%).
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên nhấn nút "Thêm thu nhập" tại màn hình Tổng quan ví.
        \item Hệ thống hiển thị biểu mẫu nhập thông tin gồm: Số tiền thu nhập, nguồn thu nhập, mô tả chi tiết, và tùy chọn phân bổ.
        \item Sinh viên nhập số tiền (ví dụ: $10.000.000$ VNĐ), chọn danh mục nguồn thu (ví dụ: "Lương làm thêm"), viết mô tả và bật tùy chọn "Tự động phân bổ".
        \item Sinh viên nhấn nút "Xác nhận".
        \item Mobile App gửi yêu cầu \texttt{POST /finance/transactions/distribute-income} kèm thông tin số tiền và mô tả lên Backend.
        \item Backend nhận yêu cầu, truy vấn bảng \texttt{money\_jars} để lấy cấu hình tỷ lệ phần trăm của sinh viên hiện tại (ví dụ: 55\%, 10\%, 10\%, 10\%, 10\%, 5\%).
        \item Backend tính toán số tiền phân bổ cụ thể cho từng hũ tài chính, đồng thời kiểm tra tổng số tiền đã phân bổ cho các hũ phải khớp chính xác với số tiền thu nhập ban đầu.
        \item Backend thực hiện đồng loạt (\texttt{Promise.all}) 6 lệnh ghi bản ghi giao dịch thu vào bảng \texttt{financial\_transactions}, mỗi lệnh cập nhật số dư của một hũ tương ứng. Đây hiện là các lệnh ghi độc lập chứ chưa được bọc trong một Database Transaction chung -- một điểm cần bổ sung để đảm bảo an toàn tuyệt đối nếu có lỗi xảy ra giữa chừng.
        \item Sau khi toàn bộ 6 lệnh ghi hoàn thành, Backend phản hồi kết quả về Mobile App.
        \item Mobile App cập nhật trạng thái dữ liệu (qua React Query) và hiển thị thông báo phân bổ tiền thành công kèm chi tiết số tiền từng lọ nhận được.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Số tiền không hợp lệ:} Tại bước 3, sinh viên nhập số tiền bằng 0 hoặc số âm. Hệ thống hiển thị lỗi ngay tại giao diện di động: "Số tiền thu nhập phải lớn hơn 0" và không cho phép gửi yêu cầu.
        \item \textit{Ngoại lệ 2 - Tổng số tiền phân bổ không khớp:} Tại bước 6-7, nếu client tự gửi kèm danh sách phân bổ chi tiết theo từng hũ (\texttt{jarAllocations}) mà tổng số tiền của danh sách này không bằng đúng số tiền thu nhập ban đầu, Backend hủy tiến trình, trả về lỗi. Mobile App hiển thị thông báo: "Số tiền phân bổ không khớp với số tiền thu nhập, vui lòng kiểm tra lại". \textit{Lưu ý:} hiện tại Backend chưa kiểm tra độc lập việc tổng tỷ lệ phần trăm (\texttt{percentage}) của 6 hũ có bằng 100\% hay không trước khi phân bổ theo tỷ lệ mặc định -- nếu dữ liệu tỷ lệ hũ của người dùng bị lệch 100\% do thao tác trước đó, hệ thống vẫn phân bổ tiền theo đúng tỷ lệ đang lưu mà không cảnh báo. Đây là một khoảng trống cần bổ sung validation trong các phiên bản kế tiếp.
        \item \textit{Ngoại lệ 3 - Lỗi kết nối cơ sở dữ liệu giữa chừng:} Tại bước 8, nếu một trong 6 lệnh ghi thất bại do mất kết nối cơ sở dữ liệu, các bản ghi giao dịch đã ghi thành công trước đó (nếu có) sẽ không được rollback tự động do chưa dùng chung một Database Transaction. Backend trả về lỗi HTTP 500 Internal Server Error, Mobile App hiển thị thông báo "Hệ thống bận, vui lòng thử lại sau" và khuyến nghị người dùng kiểm tra lại lịch sử giao dịch trước khi thực hiện lại thao tác.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Số dư khả dụng của từng hũ được cập nhật tăng tương ứng. 6 bản ghi giao dịch thu nhập mới được lưu vào cơ sở dữ liệu PostgreSQL. Hệ thống đồng bộ dữ liệu số dư mới lên màn hình Tổng quan ví.
\end{itemize}

\noindent \textbf{Chức năng 2: Thiết lập ngưỡng cảnh báo chi tiêu theo hũ}
\begin{itemize}
    \item \textbf{Tên chức năng:} Thiết lập ngưỡng cảnh báo chi tiêu cho từng hũ tài chính.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Cho phép sinh viên cấu hình một ngưỡng cảnh báo cho từng hũ tài chính theo một trong hai chế độ độc lập: (a) cảnh báo khi số dư hũ giảm xuống dưới một số tiền cố định, hoặc (b) cảnh báo khi tỷ lệ đã chi vượt quá một phần trăm nhất định. Khác với mô hình "hạn mức ngân sách theo tháng" ban đầu, cấu hình này không giới hạn theo một tháng/năm cụ thể mà áp dụng liên tục cho tới khi sinh viên thay đổi lại.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập thành công vào Mobile App và đã có ít nhất một hũ tài chính.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập vào màn hình "Chi tiết hũ" của một hũ bất kỳ (ví dụ: hũ Hưởng thụ - \texttt{enjoyment}).
        \item Sinh viên nhấn chọn "Cài đặt cảnh báo chi tiêu".
        \item Hệ thống hiển thị giao diện với hai lựa chọn độc lập có thể bật/tắt riêng: "Cảnh báo theo số dư tối thiểu" (nhập số tiền, ví dụ $200.000$ VNĐ) và "Cảnh báo theo phần trăm đã chi" (nhập số nguyên từ 0-100, ví dụ 80\%).
        \item Sinh viên bật một hoặc cả hai lựa chọn, nhập giá trị tương ứng và nhấn "Lưu".
        \item Mobile App gửi request \texttt{PUT /finance/jars/\{id\}/notification-settings} lên Backend.
        \item Backend xác thực giá trị đầu vào (phần trăm phải là số nguyên trong khoảng 0-100, số tiền phải không âm), sau đó tạo mới hoặc cập nhật bản ghi duy nhất tương ứng với hũ trong bảng \texttt{jar\_notification\_settings}.
        \item Backend phản hồi thành công về Mobile App.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Giá trị nhập không hợp lệ:} Nếu sinh viên nhập phần trăm ngoài khoảng 0-100 hoặc số tiền âm, Backend từ chối yêu cầu ở tầng validate DTO và trả về lỗi 400 Bad Request. Mobile App hiển thị thông báo lỗi tương ứng ngay tại form nhập liệu.
        \item \textit{Lưu ý về giới hạn hiện tại:} API cập nhật ngưỡng cảnh báo hiện chưa trả về kèm dữ liệu "đã chi bao nhiêu / ngưỡng bao nhiêu" đã tính sẵn, nên Mobile App chưa thể hiển thị trực tiếp thanh tiến trình (progress bar) chỉ từ phản hồi của API này mà cần tự tính toán thêm từ dữ liệu giao dịch. Đồng thời, nhánh cảnh báo theo phần trăm hiện dựa trên trường \texttt{spentAmount}/\texttt{amount} của bảng \texttt{budgets} (jar tags, xem Chương 3) — trường này chưa được các luồng nghiệp vụ hiện hành cập nhật chủ động, nên trong thực tế nhánh cảnh báo theo phần trăm có thể chưa hoạt động ổn định như nhánh cảnh báo theo số dư tối thiểu.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Bản ghi ngưỡng cảnh báo được cập nhật hoặc tạo mới trong bảng \texttt{jar\_notification\_settings}. Từ giao dịch chi tiêu tiếp theo trên hũ đó, hệ thống sẽ kiểm tra ngưỡng này ngay trong luồng tạo giao dịch (đồng bộ, không cần chờ job quét định kỳ) và gửi cảnh báo qua hàng đợi thông báo đẩy nếu vượt ngưỡng.
\end{itemize}

\subsection{Trợ lý AI tài chính}
\noindent \textbf{Chức năng 1: Trò chuyện cùng Trợ lý AI}
\begin{itemize}
    \item \textbf{Tên chức năng:} Trò chuyện và tư vấn cùng Trợ lý tài chính ảo.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên đặt câu hỏi bằng ngôn ngữ tự nhiên về tình hình tài chính cá nhân hoặc tiêu chí học bổng. Trợ lý AI sẽ phân tích, gọi các công cụ truy vấn dữ liệu thực tế và trả lời sinh viên dưới dạng dòng dữ liệu truyền liên tục (stream SSE).
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập và thiết bị di động có kết nối internet hoạt động ổn định.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Trợ lý AI" và nhập câu hỏi (ví dụ: "Tháng này mình đã tiêu bao nhiêu tiền ăn từ hũ thiết yếu?").
        \item Sinh viên nhấn nút gửi câu hỏi.
        \item Mobile App tạo kết nối Server-Sent Events (SSE) tới endpoint \texttt{/ai/chat/stream} của Backend NestJS (phân biệt với endpoint \texttt{/ai/chat} trả JSON không streaming).
        \item Backend NestJS xác thực token người dùng, sau đó chuyển tiếp yêu cầu sang FastAPI AI Service tại endpoint \texttt{/api/v1/chat/stream}.
        \item FastAPI AI Service khởi chạy LangChain Agent, gửi câu hỏi kèm lịch sử trò chuyện ngắn và mã định danh sinh viên.
        \item Agent phân tích ý định câu hỏi và nhận diện cần dữ liệu chi tiêu thực tế. Agent quyết định gọi tool \texttt{search\_transactions} (hoặc \texttt{get\_recent\_transactions} nếu chỉ cần các giao dịch gần nhất) với tham số truy vấn là tháng hiện tại và danh mục "ăn uống/thực phẩm".
        \item Agent gửi yêu cầu truy vấn ngược lại Backend NestJS để lấy dữ liệu giao dịch từ PostgreSQL.
        \item Backend NestJS trả về danh sách giao dịch ăn uống của sinh viên trong tháng.
        \item Agent nhận dữ liệu, tính toán tổng tiền, soạn thảo văn bản nhận định tài chính và gửi từng token chữ được sinh ra từ Gemini API về FastAPI AI Service.
        \item FastAPI AI Service lập tức truyền các token chữ này về Backend NestJS dưới dạng stream SSE, Backend NestJS tiếp tục đẩy stream này về ứng dụng di động của sinh viên.
        \item Mobile App nhận từng token chữ và hiển thị trực tiếp lên màn hình dưới dạng chữ chạy thời gian thực.
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Gọi API Gemini bị lỗi hoặc hết hạn mức (Quota):} Tại bước 9, Gemini API trả về lỗi giới hạn lượt gọi hoặc quá tải. FastAPI AI Service sẽ bắt ngoại lệ này, sinh một câu trả lời mặc định thân thiện: "Xin lỗi bạn, kết nối tới máy chủ trí tuệ nhân tạo hiện tại đang gián đoạn. Vui lòng thử lại sau giây lát!" và kết thúc stream.
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Phản hồi từ AI được truyền tải và hiển thị đầy đủ lên màn hình trò chuyện của sinh viên.
\end{itemize}

\noindent \textbf{Chức năng 2: Phân loại giao dịch bằng AI}
\begin{itemize}
    \item \textbf{Tên chức năng:} Phân loại giao dịch chi tiêu tự động bằng AI.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên chỉ cần gõ một câu mô tả ngắn gọn về khoản thu chi bằng ngôn ngữ tự nhiên, AI sẽ tự động phân tích để trích xuất số tiền, phân loại hũ tài chính phù hợp và gợi ý danh mục chi tiết để sinh viên xác nhận lưu nhanh.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập và đang mở biểu mẫu thêm giao dịch nhanh.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên mở ô nhập nhanh giao dịch và gõ câu lệnh: "mua sach 150k bang tien mat".
        \item Sinh viên nhấn nút "Phân tích bằng AI".
        \item Mobile App gọi API \texttt{POST /api/ai/6jars/classify} kèm câu mô tả văn bản thô.
        \item Backend NestJS chuyển tiếp payload tới FastAPI AI Service.
        \item FastAPI sử dụng Few-shot Prompting kết hợp mô hình Gemini để phân tích ngữ nghĩa câu thô.
        \item AI nhận diện: hành động là "chi tiêu" (\texttt{expense}), số tiền là $150.000$, hũ tài chính phù hợp là hũ "Giáo dục" (\texttt{education}), danh mục đề xuất là "Sách vở/Học cụ".

        \item \textit{Luồng phụ - Người dùng điều chỉnh lại gợi ý sai của AI:} Tại bước 8, nếu AI phân loại sai hũ (ví dụ gợi ý hũ Hưởng thụ thay vì Nhu cầu thiết yếu), sinh viên có thể tự tay chọn lại hũ Nhu cầu thiết yếu trên giao diện.     \item \textbf{Kết quả (Postconditions):} Các trường thông tin được điền tự động và giao dịch được tạo chính xác trong DB.

Nếu vượt giới hạn, ứng dụng hiển thị thông báo: "Tài liệu vượt quá dung lượng cho phép. Vui lòng nén file và thử lại" và không cho phép tiếp tục.

\subsection{Quản lý dữ liệu học thuật}
\noindent \textbf{Chức năng 1: Quản lý bảng điểm cá nhân}
\begin{itemize}
    \item \textbf{Tên chức năng:} Quản lý bảng điểm cá nhân và tự động tính toán GPA.
    \item \textbf{Tác nhân (Actor):} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên tự cập nhật điểm số các môn học đã hoàn thành của mình trên ứng dụng di động. Hệ thống sẽ tự động tính toán điểm trung bình tích lũy GPA dựa trên số tín chỉ của từng môn học được tham chiếu từ danh mục môn học chuẩn hóa của trường.
    \item \textbf{Điều kiện trước (Preconditions):} Sinh viên đã đăng nhập và đã cấu hình thông tin trường đại học trong hồ sơ cá nhân.
    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Góc học tập" và chọn tab "Bảng điểm".
        \item Hệ thống hiển thị danh sách các môn học đã nhập điểm kèm số tín chỉ và điểm số cụ thể.
        \item Sinh viên nhấn nút "Thêm điểm môn học".
        \item Hệ thống hiển thị biểu mẫu bao gồm: Danh sách môn học chuẩn, Ô nhập điểm số.
        \item Sinh viên gõ tìm kiếm và chọn môn học (ví dụ: "Nhập môn Lập trình - 4 tín chỉ"), nhập điểm số đạt được (ví dụ: 8.5 hoặc điểm chữ A) và nhấn "Lưu".
        \item Mobile App gửi yêu cầu \texttt{POST /student/grades} lên Backend.
        \item Backend xác thực môn học thuộc về trường đại học sinh viên đang học, kiểm tra điểm số có nằm trong giới hạn thang điểm của môn học không.
        \item Backend lưu bản ghi điểm môn học vào bảng \texttt{student\_grades} và phản hồi thành công về Mobile App. Điểm GPA tích lũy \textbf{không} được ghi đè trực tiếp vào bảng hồ sơ học thuật \texttt{student\_academic\_profiles} ngay tại bước này.
        \item Khi sinh viên mở màn hình Dashboard bảng điểm (\texttt{GET /student/dashboard}), Backend mới truy vấn toàn bộ các môn học có điểm của sinh viên, lấy số tín chỉ và điểm số của từng môn để tính GPA tích lũy có trọng số theo công thức sau, tính trực tiếp tại thời điểm truy vấn (không lưu lại giá trị này vào cơ sở dữ liệu):
        $$\text{GPA} = \frac{\sum (\text{Điểm môn } i \times \text{Số tín chỉ môn } i)}{\sum (\text{Số tín chỉ})}$$
        \item Mobile App hiển thị bảng điểm được cập nhật mới kèm chỉ số GPA tích lũy tính theo thời gian thực (ví dụ: GPA: 3.42 / 4.0).
    \end{enumerate}
    \item \textbf{Luồng ngoại lệ (Exception Flows):}
    \begin{itemize}
        \item \textit{Ngoại lệ 1 - Điểm số không hợp lệ:} Tại bước 5, sinh viên nhập điểm là 11 (vượt quá thang điểm 10) hoặc nhập số âm. Hệ thống hiển thị thông báo lỗi ngay trên giao diện nhập: "Điểm số nhập vào không hợp lệ. Vui lòng nhập điểm trong khoảng từ 0 đến 10".
    \end{itemize}
    \item \textbf{Kết quả (Postconditions):} Bảng điểm và GPA tích lũy được cập nhật tức thời trên màn hình Dashboard học tập.
\end{itemize}

\noindent \textbf{Chức năng 2: Nhập điểm hàng loạt từ file}

    \item \textbf{Luồng chính (Basic Flow):}
    \begin{enumerate}
        \item Sinh viên truy cập màn hình "Bảng điểm" và chọn chức năng "Nhập điểm từ tệp tin".
        \item Hệ thống hiển thị hộp thoại yêu cầu chọn tệp tin và hướng dẫn định dạng file mẫu.

        \item Mobile App thực hiện đọc nội dung tệp tin cục bộ, sử dụng bộ Parser phân tích cú pháp từng dòng của tệp để trích xuất ra các cột thông tin: Mã môn học, Điểm số.
        \item Hệ thống hiển thị danh sách các môn học và điểm số đã parse được lên màn hình xem trước để sinh viên rà soát lại.

        \item Mobile App gửi toàn bộ danh sách các cặp giá trị (Mã môn học, Điểm số) đã được parse trong một lần gọi duy nhất tới cùng endpoint \texttt{POST /student/grades} (không phải một endpoint \texttt{/bulk} riêng), với phần thân request là một mảng nhiều môn học/điểm số.
        \item Backend kiểm tra sự tồn tại của các mã môn học trong cơ sở dữ liệu Master Data của trường đại học mà sinh viên đăng ký.
        \item Backend lặp qua từng phần tử trong mảng để xóa điểm cũ của những môn học trùng lặp (nếu có) và thêm mới các bản ghi điểm môn học vào bảng \texttt{student\_grades}.

    \item \textbf{Luồng ngoại lệ (Exception Flows):}

Hệ thống dừng tiến trình và hiển thị thông báo lỗi: "Tệp tin không đúng định dạng. Vui lòng tải lên file CSV hoặc TXT theo đúng cấu trúc hướng dẫn".
        \item \textit{Ngoại lệ 2 - Mã môn học không tồn tại trong Master Data:} Tại bước 8, Backend phát hiện có mã môn học trong danh sách gửi lên không tồn tại trong danh mục môn học chuẩn của trường đại học đó. Backend sẽ trả về danh sách các môn học bị lỗi. Mobile App hiển thị hộp thoại thông báo: "Phát hiện một số môn học có mã không hợp lệ.     \item \textbf{Kết quả (Postconditions):} Danh sách môn học và điểm số trích xuất thành công được ghi nhận hàng loạt vào cơ sở dữ liệu, GPA tích lũy được cập nhật lại.
\end{itemize}
