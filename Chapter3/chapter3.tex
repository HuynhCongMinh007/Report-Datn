\chapter{Phân tích và thiết kế}
\label{ch:phan-tich-thiet-ke}

Chương này trình bày quá trình phân tích yêu cầu và thiết kế hệ thống Student360. Mục~3.1 xác định tác nhân và các nhóm Use Case chính. Mục~3.2 đặc tả chi tiết các chức năng nghiệp vụ theo từng phân hệ. Mục~3.3 trình bày kiến trúc tổng quan bốn tầng. Mục~3.4 đến 3.7 thiết kế chi tiết từng tầng: Frontend, Backend, AI và dữ liệu.

\section{Phân tích yêu cầu hệ thống}
\label{sec:phan-tich-yeu-cau}
Như đã giới thiệu tại Chương 1, hệ thống Student360 gồm 4 phân hệ chức năng cốt lõi: Quản lý tài chính cá nhân, Trợ lý tài chính ảo AI, Quản lý học bổng trực tuyến và Quản lý học thuật/Bảng điểm. Mục này xác định tác nhân tham gia và sơ đồ Use Case của từng phân hệ; đặc tả chi tiết từng chức năng được trình bày tại Mục~3.2.

\section{Luồng nghiệp vụ chính}
Mục này đặc tả chi tiết từng chức năng nghiệp vụ theo bốn phân hệ: tài chính cá nhân (Mục~3.2.1), trợ lý AI tài chính (Mục~3.2.2), học bổng (Mục~3.2.3) và học thuật/bảng điểm (Mục~3.2.4). Mỗi chức năng được gán một mã Use Case theo quy ước tiền tố phân hệ (FIN, AI, SCH, ACA) cộng số thứ tự, ví dụ FIN-UC01 hay SCH-UC03.

\subsection{Luồng nghiệp vụ quản lý tài chính cá nhân}

\noindent\textbf{FIN-UC01 -- Cấu hình tỷ lệ 6 lọ tài chính}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên tùy chỉnh tỷ lệ phân bổ giữa các hũ tài chính (6 hũ mặc định hoặc hũ tự thêm), làm cơ sở cho chức năng phân bổ thu nhập tự động (FIN-UC03).
    \item \textbf{Luồng chính:} Mở màn hình cấu hình hũ, xem tỷ lệ hiện tại $\rightarrow$ điều chỉnh tỷ lệ từng hũ, có thể thêm hoặc gỡ hũ tùy chỉnh $\rightarrow$ lưu; hệ thống kiểm tra tổng tỷ lệ đúng 100\% rồi cập nhật cấu hình.
    \item \textbf{Ngoại lệ:} Tổng tỷ lệ khác 100\% hoặc chỉnh sửa hũ không thuộc sở hữu đều bị từ chối.
    \item \textbf{Kết quả:} Cấu hình mới áp dụng ngay cho lần phân bổ thu nhập tiếp theo.
\end{itemize}

\noindent\textbf{FIN-UC02 -- Ghi nhận giao dịch thủ công}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên tự nhập một giao dịch Thu, Chi hoặc Chuyển khoản giữa các hũ, không qua bước phân tích của AI -- khác với chức năng phân loại tự động (AI-UC02).
    \item \textbf{Luồng chính:} Chọn loại giao dịch, nhập số tiền, hũ liên quan, nhãn và ghi chú $\rightarrow$ lưu; hệ thống xác thực dữ liệu và cập nhật lịch sử giao dịch cùng số dư hũ liên quan.
    \item \textbf{Ngoại lệ:} Số tiền không hợp lệ (bằng 0 hoặc âm) hoặc hũ không thuộc sở hữu đều bị từ chối.
    \item \textbf{Kết quả:} Giao dịch được lưu, số dư cập nhật ngay lập tức. Thao tác sửa/xóa giao dịch được đánh giá riêng tại Chương~\ref{ch:hien-thuc-kiem-thu-danh-gia}.
\end{itemize}

\noindent\textbf{FIN-UC03 -- Phân bổ thu nhập tự động}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Ghi nhận một khoản thu mới và tự động chia vào các hũ theo tỷ lệ đã cấu hình (FIN-UC01).
    \item \textbf{Luồng chính:} Nhập số tiền, nguồn thu, bật tùy chọn tự động phân bổ $\rightarrow$ xác nhận; hệ thống tính số tiền cho từng hũ theo tỷ lệ, kiểm tra tổng khớp với số tiền gốc, rồi ghi nhận đồng loạt các giao dịch thu tương ứng và cập nhật số dư từng hũ.
    \item \textbf{Ngoại lệ:} Số tiền không hợp lệ hoặc tổng phân bổ không khớp số tiền gốc đều bị từ chối.
    \item \textbf{Kết quả:} Số dư từng hũ tăng tương ứng; toàn bộ giao dịch được lưu và đồng bộ lên màn hình tổng quan.
    \item \textbf{Toàn vẹn số dư:} Số dư mỗi hũ được tính lại từ lịch sử giao dịch mỗi khi có thay đổi, giúp tự phục hồi nếu giá trị lưu sẵn bị sai lệch. Khi xóa một hũ tùy chỉnh còn số dư, hệ thống bắt buộc chọn hũ nhận lại tiền và tạo một cặp giao dịch bù trừ để tổng tài sản của sinh viên không đổi. Cơ chế giao dịch/idempotency liên quan trình bày tại Mục~3.5.3.
\end{itemize}

\noindent\textbf{FIN-UC04 -- Thiết lập ngưỡng cảnh báo chi tiêu theo hũ}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Cấu hình ngưỡng cảnh báo cho từng hũ theo số dư tối thiểu hoặc phần trăm đã chi, áp dụng liên tục tới khi sinh viên thay đổi lại.
    \item \textbf{Luồng chính:} Mở chi tiết hũ $\rightarrow$ bật một hoặc cả hai loại cảnh báo, nhập giá trị $\rightarrow$ lưu; hệ thống xác thực và lưu cấu hình.
    \item \textbf{Ngoại lệ:} Giá trị nhập ngoài khoảng hợp lệ bị từ chối.
    \item \textbf{Kết quả:} Từ giao dịch chi tiêu tiếp theo, hệ thống kiểm tra ngưỡng ngay khi tạo giao dịch và gửi cảnh báo đẩy nếu vượt.
    \item \textbf{Cảnh báo tự động:} Ngoài ngưỡng do sinh viên tự đặt, hệ thống còn một tác vụ định kỳ hằng ngày tự phát hiện giao dịch bất thường; một khóa định danh duy nhất đảm bảo không gửi trùng cảnh báo nếu tác vụ chạy lại.
\end{itemize}

\noindent\textbf{FIN-UC05 -- Lên lịch giao dịch định kỳ}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Thiết lập một giao dịch lặp lại tự động (hằng ngày, hằng tuần hoặc hằng tháng), hệ thống tự tạo giao dịch đúng thời điểm mà không cần thao tác thủ công mỗi lần.
    \item \textbf{Luồng chính:} Tạo lịch mới (tên, số tiền, loại giao dịch, tần suất, cách phân bổ, tùy chọn nhắc) $\rightarrow$ lưu; hệ thống lên lịch tác vụ nền, tới đúng thời điểm tự tạo giao dịch tương ứng và đặt lại chu kỳ kế tiếp.
    \item \textbf{Ngoại lệ:} Lịch bị xóa trong lúc tác vụ đang chờ thì bị bỏ qua; cấu hình phân bổ tùy chỉnh không hợp lệ bị từ chối khi tạo.
    \item \textbf{Kết quả:} Hệ thống tự tạo giao dịch đúng chu kỳ đã cấu hình; sinh viên có thể bật/tắt, sửa, xóa hoặc kích hoạt thủ công bất kỳ lúc nào.
\end{itemize}

\subsection{Luồng nghiệp vụ trợ lý AI tài chính}

\noindent\textbf{AI-UC01 -- Trò chuyện cùng Trợ lý AI}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên đặt câu hỏi tự nhiên về tài chính cá nhân hoặc học bổng, nhận phản hồi theo thời gian thực.
    \item \textbf{Luồng chính:} Gửi câu hỏi qua kênh trò chuyện $\rightarrow$ hệ thống xác định ý định câu hỏi, tác nhân AI truy vấn dữ liệu thực tế nếu cần, rồi trả lời dần theo thời gian thực (kiến trúc chi tiết tại Mục~3.6.1, minh họa ở Hình~\ref{fig:ai-flow}).
    \item \textbf{Ngoại lệ:} Nếu mô hình ngôn ngữ lớn lỗi hoặc quá tải, hệ thống trả lời mặc định xin lỗi và kết thúc phiên.
    \item \textbf{Kết quả:} Câu trả lời hiển thị đầy đủ trên màn hình trò chuyện.
\end{itemize}

\noindent\textbf{AI-UC02 -- Phân loại giao dịch bằng AI}
\begin{itemize}
    \item \textbf{Tác nhân:} Sinh viên.
    \item \textbf{Mô tả:} Sinh viên gõ một mô tả ngắn về khoản thu chi bằng ngôn ngữ tự nhiên; AI trích số tiền, đề xuất hũ và danh mục phù hợp để sinh viên xác nhận lưu nhanh.
    \item \textbf{Luồng chính:} Nhập mô tả, yêu cầu phân tích bằng AI $\rightarrow$ hệ thống trích xuất số tiền, loại giao dịch, hũ và danh mục đề xuất, điền sẵn vào biểu mẫu $\rightarrow$ sinh viên kiểm tra và xác nhận lưu.
    \item \textbf{Ngoại lệ:} Nếu AI gợi ý sai, sinh viên tự chọn lại; lựa chọn sửa được ghi nhớ riêng cho tài khoản đó để cá nhân hóa gợi ý về sau (không làm thay đổi mô hình dùng chung cho các tài khoản khác).
    \item \textbf{Kết quả:} Giao dịch được tạo với thông tin chính xác trong cơ sở dữ liệu.
\end{itemize}

\subsection{Luồng nghiệp vụ học bổng}

\noindent\textbf{SCH-UC01 -- Nhà tài trợ quản lý chương trình học bổng}
\begin{itemize}
    \item \textbf{Tác nhân:} Nhà tài trợ.
    \item \textbf{Mô tả:} Tạo, chỉnh sửa hoặc gỡ một chương trình học bổng: thông tin chung, giá trị, tiêu chí xét duyệt, thời hạn nộp và danh sách tài liệu yêu cầu.
    \item \textbf{Luồng chính:} Tạo học bổng mới với đầy đủ thông tin $\rightarrow$ thêm tài liệu yêu cầu và điều kiện xét duyệt $\rightarrow$ kích hoạt công khai để sinh viên xem và ứng tuyển (SCH-UC02); có thể sửa hoặc gỡ học bổng đã tạo.
    \item \textbf{Ngoại lệ:} Thiếu thông tin bắt buộc bị từ chối; tài khoản không đúng vai trò không truy cập được màn hình quản lý.
    \item \textbf{Kết quả:} Học bổng được tạo, sửa hoặc xóa tương ứng; nếu kích hoạt, hiển thị cho sinh viên.
\end{itemize}

\section{Kiến trúc tổng quan}

\subsection{Kiến trúc bốn tầng và nguyên lý thiết kế}
Kiến trúc Student360 hướng tới bốn mục tiêu: (1) tách rõ trách nhiệm giữa các thành phần; (2) cho phép phát triển, kiểm thử, vận hành độc lập; (3) cho phép nhiều thành viên làm việc song song mà ít xung đột mã nguồn, phù hợp cách phân công theo cặp tại Chương 4; (4) cô lập rủi ro và chi phí vận hành của AI khỏi luồng nghiệp vụ lõi.

Hệ thống theo mô hình \textbf{Client--Server phân tán bốn tầng}: tầng trình bày, tầng ứng dụng (Backend), tầng AI (AI Service) và tầng dữ liệu. Tầng AI tách riêng khỏi tầng ứng dụng vì đặc tính vận hành khác hẳn nghiệp vụ thông thường của Backend (độ trễ gọi mô hình ngôn ngữ lớn, phụ thuộc nhà cung cấp bên thứ ba, chi phí theo token). Mỗi thành phần nằm trong một repository Git riêng (đa-repository) và chỉ giao tiếp qua mạng (HTTP REST, SSE, hàng đợi thông điệp), phù hợp cách làm việc theo cặp của nhóm 6 thành viên.

\begin{itemize}
    \item \textbf{Tầng trình bày:} Mobile App và Web Admin Portal.
    \item \textbf{Tầng ứng dụng:} Backend -- xác thực, phân quyền, xử lý nghiệp vụ, điều phối dữ liệu.
    \item \textbf{Tầng AI:} AI Service -- xử lý ngôn ngữ tự nhiên và các tác vụ AI.
    \item \textbf{Tầng dữ liệu:} PostgreSQL và Redis -- lưu trữ nghiệp vụ và hàng đợi bất đồng bộ.
\end{itemize}

\section{Thiết kế Frontend}

\subsection{Nền tảng di động}
Ứng dụng Mobile (Expo/React Native) là kênh tương tác chính của sinh viên: ghi nhận thu chi, trò chuyện với AI, ứng tuyển học bổng, quản lý bảng điểm và nhận thông báo đẩy. Nền tảng này được chọn vì cho phép phát triển đồng thời iOS và Android từ một mã nguồn, phù hợp quy mô và thời gian của đồ án; thư viện giao diện Tamagui giữ phong cách hiển thị đồng bộ trên thiết bị di động.

Ứng dụng chia thành ba tầng: tầng giao diện và định tuyến (theo cấu trúc tệp tin, phản ánh trực tiếp cấu trúc điều hướng), tầng trạng thái (tách trạng thái phiên/giao diện cục bộ khỏi trạng thái dữ liệu nghiệp vụ từ máy chủ), và tầng giao tiếp API (gom việc gọi Backend vào một lớp client duy nhất, tự đính kèm và làm mới token xác thực). Dữ liệu nghiệp vụ đồng bộ hai chiều với Backend qua cơ chế làm mới bộ đệm tự động mỗi khi có thay đổi, giúp số dư và lịch sử giao dịch luôn đúng với dữ liệu mới nhất mà không cần người dùng tải lại thủ công.

\section{Thiết kế Backend}

\subsection{Kiến trúc hệ thống Backend}
Backend là trung tâm điều phối của hệ thống: xác thực, phân quyền, xử lý nghiệp vụ, là điểm truy cập duy nhất tới cơ sở dữ liệu, và là cổng trung gian cho các yêu cầu cần AI Service. Backend theo \textbf{kiến trúc phân lớp}: một dịch vụ duy nhất, module hóa theo miền nghiệp vụ, không tách thành nhiều dịch vụ độc lập kiểu Microservices. Lựa chọn này phù hợp quy mô nhóm 6 thành viên và thời gian đồ án có hạn: module hóa trong một dịch vụ đã đủ tách biệt trách nhiệm, mà không phải gánh chi phí vận hành nhiều dịch vụ độc lập.

Một yêu cầu đi qua Backend lần lượt qua các tầng:
\begin{itemize}
    \item \textbf{Tiếp nhận và xác thực:} xác thực danh tính, kiểm tra quyền truy cập theo vai trò.
    \item \textbf{Chuẩn hóa và giám sát:} chuẩn hóa phản hồi, ghi log phục vụ theo dõi và gỡ lỗi.
    \item \textbf{Xử lý nghiệp vụ:} hiện thực logic nghiệp vụ, độc lập với cách tiếp nhận yêu cầu hay lưu trữ dữ liệu.
    \item \textbf{Truy xuất dữ liệu:} đọc/ghi cơ sở dữ liệu, cô lập tầng nghiệp vụ khỏi chi tiết truy vấn.
    \item \textbf{Đẩy tác vụ nền:} với tác vụ không cần phản hồi tức thời, đẩy vào hàng đợi thay vì xử lý đồng bộ.
\end{itemize}

\subsection{Framework và công nghệ}
Backend dùng NestJS \cite{nestjs-docs} vì framework có sẵn cơ chế module hóa rõ ràng, khớp với mô hình phân lớp ở trên; hệ sinh thái TypeScript cũng giảm lỗi khi nhiều thành viên phát triển song song. Truy xuất dữ liệu quan hệ dùng ORM có sẵn của framework; tác vụ nền và bất đồng bộ dùng cơ chế hàng đợi trên nền Redis.

\subsection{Các cơ chế bảo vệ nghiệp vụ}
Backend thiết kế cơ chế bảo vệ nghiệp vụ theo hai lớp. Lớp thứ nhất là xác thực và phân quyền: mọi yêu cầu API đều đi qua JWT, Backend lấy danh tính người dùng từ token đã xác thực thay vì tin vào định danh do client truyền lên; phạm vi thao tác được giới hạn theo vai trò sinh viên, nhà tài trợ hoặc quản trị viên. Lớp thứ hai là bảo vệ toàn vẹn dữ liệu cho các nghiệp vụ có nhiều bước ghi dữ liệu.

Ngoài xác thực/phân quyền và chuẩn hóa dữ liệu đầu vào đã nói ở Mục~3.5.1, Backend dùng thêm ba cơ chế cho các thao tác nhạy cảm với tính nhất quán dữ liệu:

\begin{itemize}
    \item \textbf{Tính nhất quán khi thao tác nhiều lọ tài chính:} một lần chuyển tiền giữa hai lọ tạo đồng thời hai bản ghi (chi ở lọ nguồn, thu ở lọ đích); nếu một bước lỗi, toàn bộ được hoàn tác để không xảy ra trạng thái chỉ ghi được một phía. Số dư mỗi lọ được tính lại từ lịch sử giao dịch sau khi ghi thành công, thay vì chỉ tin vào giá trị tạm trên giao diện.
    \item \textbf{Không phát sinh hồ sơ học bổng trùng:} hồ sơ ứng tuyển được ràng buộc duy nhất theo từng cặp sinh viên -- học bổng; việc cập nhật hồ sơ cá nhân, tài liệu và lịch sử trạng thái được xử lý như một thao tác trọn vẹn để tránh dữ liệu dở dang nếu lỗi giữa chừng.
    \item \textbf{Idempotency cho tác vụ có thể chạy lại:} các tác vụ nền có thể chạy lại (ví dụ quét cảnh báo bất thường định kỳ) dùng một khóa định danh duy nhất để loại bỏ trùng lặp, đảm bảo không tạo ra bản ghi hay thông báo trùng cho người dùng.
\end{itemize}

\section{Thiết kế AI}

\subsection{AI Agent tư vấn}
AI Service dùng FastAPI \cite{fastapi-docs} kết hợp LangChain \cite{langchain-docs} để xây dựng tác nhân AI, kết nối tới mô hình ngôn ngữ lớn Gemini \cite{gemini-api}. Dịch vụ này cung cấp ba năng lực: tư vấn tài chính qua hội thoại, phân loại giao dịch từ mô tả tự nhiên, và gợi ý học bổng phù hợp hồ sơ sinh viên. AI Service được tách thành thành phần độc lập để cô lập độ trễ và chi phí gọi mô hình ngôn ngữ lớn khỏi luồng nghiệp vụ lõi của Backend; client không gọi trực tiếp AI Service mà luôn đi qua Backend với một token nội bộ thời hạn ngắn. AI Service dùng Python/FastAPI thay vì NestJS như Backend vì hệ sinh thái thư viện AI/LLM của Python phong phú hơn, và LangChain có sẵn các thành phần cho tác nhân tool-calling cũng như tích hợp đa nhà cung cấp mô hình ngôn ngữ.

Khi sinh viên đặt câu hỏi hoặc nhập mô tả giao dịch, yêu cầu được Backend xác thực rồi chuyển sang AI Service. Hệ thống trước tiên phân loại ý định câu hỏi để xác định phạm vi xử lý, sau đó một tác nhân AI suy luận theo mô hình lặp truy vấn--hành động (ReAct) \cite{yao2022-react}, gọi các công cụ nghiệp vụ khi cần dữ liệu thực tế bằng cách gọi ngược lại Backend. Kết quả được trả về dưới dạng luồng liên tục, hiển thị dần trên ứng dụng theo thời gian thực, tương tự trải nghiệm của các trợ lý ảo thương mại (đặc tả nghiệp vụ tại Mục~3.2.2, AI-UC01).

Đối với phân loại giao dịch từ tiếng Việt tự nhiên, AI Service được thiết kế theo hướng pipeline lai. Dữ liệu đầu vào được chuẩn hóa trước khi xử lý; hệ thống ưu tiên kiểm tra các quy tắc và từ khóa cá nhân hóa đã ghi nhận từ lựa chọn trước đó của sinh viên, sau đó mới gọi mô hình ngôn ngữ lớn khi cần suy luận ngữ nghĩa. Kết quả AI không được ghi thẳng vào cơ sở dữ liệu mà chỉ dùng để điền sẵn biểu mẫu giao dịch; sinh viên phải kiểm tra và xác nhận trước khi Backend lưu giao dịch chính thức.

Các thành phần chính của AI Agent:
\begin{itemize}
    \item \textbf{Tiếp nhận yêu cầu:} xác thực token nội bộ trước khi chuyển tiếp cho tầng xử lý.
    \item \textbf{Phân loại ý định:} xác định loại câu hỏi để giới hạn đúng phạm vi ngữ cảnh và tập công cụ được phép dùng, thay vì luôn cho phép toàn bộ công cụ -- giúp giảm chi phí gọi mô hình và tăng độ chính xác.
    \item \textbf{Suy luận và gọi công cụ:} một tác nhân AI duy nhất theo mô hình tool-calling, được thu hẹp phạm vi linh hoạt theo kết quả phân loại ý định, thay vì cần nhiều tác nhân chuyên biệt phối hợp.
    \item \textbf{Công cụ nghiệp vụ:} các hàm truy vấn và tính toán cụ thể mà tác nhân AI có thể gọi khi cần dữ liệu thực tế.
    \item \textbf{Trừu tượng hóa nhà cung cấp mô hình:} cho phép chuyển đổi hoặc dự phòng linh hoạt giữa nhiều nhà cung cấp mô hình ngôn ngữ khi gặp lỗi hạn mức hoặc gián đoạn kết nối.
    \item \textbf{Xử lý nền:} tiến trình tách biệt khỏi luồng xử lý yêu cầu chính, đảm nhận tác vụ định kỳ như tổng hợp nhận định tài chính hoặc dọn dẹp phiên hội thoại không hoạt động.
\end{itemize}

Đây là nền tảng để mở rộng thành kiến trúc đa-agent thực sự khi hệ thống cần thêm nghiệp vụ AI phức tạp hơn.

\subsection{AI Matching: Đối sánh hồ sơ sinh viên với học bổng}
Bên cạnh AI Agent tư vấn, AI Service còn giải bài toán đối sánh mức độ phù hợp giữa hồ sơ sinh viên và từng chương trình học bổng, phục vụ tính năng gợi ý khi sinh viên duyệt danh sách (Mục~3.2.3, SCH-UC02).

Điểm số so khớp được tính trực tiếp tại thời điểm truy vấn, không dùng cơ sở dữ liệu vector. Các trường thông tin có giá trị trong hồ sơ sinh viên (ngành học, trường, kỹ năng, hoạt động ngoại khóa) được gộp thành một đoạn văn bản, chuẩn hóa để tiếng Việt hiển thị nhất quán, rồi so sánh với nội dung học bổng bằng ba độ đo tương đồng văn bản:
\begin{itemize}
    \item \textbf{Độ tương đồng theo từ vựng} \cite{jaccard1901} (trọng số 0.5): điểm càng cao khi hai văn bản dùng chung nhiều từ khóa.
    \item \textbf{Độ tương đồng theo cặp ký tự liền kề} \cite{cavnar1994-ngram} (trọng số 0.3): giúp nhận diện các trường hợp viết tắt hoặc sai chính tả mà so khớp theo từ không phát hiện được.
    \item \textbf{Tỷ lệ khớp chuỗi tổng thể} (trọng số 0.2): đo mức độ giống nhau tổng thể giữa hai chuỗi văn bản.
\end{itemize}
Ba độ đo được kết hợp theo trọng số cố định thành một điểm số duy nhất, hiển thị cho sinh viên dưới dạng phần trăm khớp. Cách này chi phí tính toán thấp, không cần huấn luyện mô hình riêng, phù hợp quy mô dữ liệu của đồ án; nhược điểm là chưa nắm được ngữ nghĩa sâu -- hồ sơ và học bổng dùng từ khác nhau nhưng cùng nghĩa (ví dụ "kỹ thuật phần mềm" và "công nghệ thông tin") có thể không được nhận ra là tương đương. Giới hạn này được bàn thêm ở Chương~\ref{ch:ket-luan}.

\section{Thiết kế dữ liệu}

\subsection{Lựa chọn hệ quản trị cơ sở dữ liệu}
Hệ thống dùng \textbf{PostgreSQL 15} \cite{postgresql-docs} làm nguồn dữ liệu duy nhất cho toàn bộ dữ liệu nghiệp vụ, chuẩn hóa theo mô hình quan hệ với ràng buộc khóa ngoại chặt để giữ toàn vẹn dữ liệu. \textbf{Redis 7} \cite{redis-docs} đóng vai trò làm message broker cho hàng đợi xử lý bất đồng bộ (Mục~3.3.2, 3.5.3), đồng thời được sử dụng để lưu trữ dữ liệu tạm thời (caching) nhằm hỗ trợ giảm tải cho cơ sở dữ liệu chính cũng như tối ưu hóa tốc độ phản hồi của hệ thống.

\subsection{Thiết kế mô hình dữ liệu}
Dữ liệu được tổ chức thành ba nhóm chính, thể hiện trong sơ đồ ERD ở Hình~\ref{fig:erd-finance}, \ref{fig:erd-academic} và \ref{fig:erd-scholarship}:
\begin{itemize}
    \item \textbf{Nhóm Tài chính:} các hũ tài chính, lịch sử giao dịch, lịch chuyển tiền tự động, cấu hình ngưỡng cảnh báo.
    \item \textbf{Nhóm Học thuật:} danh mục trường, chương trình đào tạo, môn học, hồ sơ học thuật và bảng điểm cá nhân.
    \item \textbf{Nhóm Học bổng:} chương trình học bổng, điều kiện xét tuyển, tài liệu yêu cầu, hồ sơ ứng tuyển, lịch sử trạng thái và kết quả đánh giá.
\end{itemize}

Nhóm Học bổng tham chiếu tới nhóm Học thuật để đối chiếu điều kiện xét duyệt (GPA tối thiểu, ngành học phù hợp...) với hồ sơ thực tế của sinh viên. Nhóm Tài chính độc lập hoàn toàn với hai nhóm còn lại, đúng ranh giới nghiệp vụ giữa quản lý dòng tiền cá nhân và quản lý học tập/học bổng. Mục tiêu tín chỉ được lưu trực tiếp trong hồ sơ học thuật của sinh viên, không tách bảng riêng; mỗi lượt chuyển trạng thái hồ sơ học bổng đều ghi lại trạng thái cũ và trạng thái mới, kết quả đánh giá của người xét duyệt được lưu riêng để phục vụ tra cứu.

Số dư hiện tại của mỗi hũ tài chính được lưu sẵn để truy vấn và hiển thị nhanh. Sau khi tạo, sửa hoặc xóa giao dịch, Backend tính lại tổng thu và tổng chi từ lịch sử giao dịch rồi cập nhật số dư và các giá trị lũy kế -- cách này vừa đọc nhanh vừa có thể khôi phục đúng số liệu từ dữ liệu gốc nếu có sai lệch. Trường hợp đặc biệt như xóa một hũ đang có số dư được nói rõ hơn tại Mục~3.2.1 (FIN-UC03).
