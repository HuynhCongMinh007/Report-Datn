\chapter{Phương pháp đề xuất và Thiết kế hệ thống}
\label{ch:phuong-phap-de-xuat}

\section{Kiến trúc tổng thể hệ thống (System Architecture)}
Hệ thống Student360 được thiết kế dựa trên mô hình monorepo đa dịch vụ nhằm đảm bảo tính phân tách trách nhiệm (Separation of Concerns), khả năng mở rộng độc lập và dễ dàng quản lý mã nguồn trong quá trình phát triển nhóm. Hệ thống bao gồm ba phân hệ chính hoạt động độc lập nhưng giao tiếp chặt chẽ với nhau thông qua mạng nội bộ và các giao thức kết nối tiêu chuẩn như HTTP REST, Server-Sent Events (SSE), và Message Queues.

\subsection{Kiến trúc Monorepo đa dịch vụ}
Cấu trúc hệ thống bao gồm bốn thành phần cốt lõi:
\begin{itemize}
    \item \textbf{Frontend Mobile Application (Expo - React Native):} Đây là ứng dụng di động đa nền tảng dành cho đối tượng sinh viên. Nó chịu trách nhiệm hiển thị giao diện người dùng, thu thập thông tin giao dịch tài chính, cung cấp cổng tương tác trò chuyện thời gian thực với trợ lý ảo, đăng ký nhận thông báo đẩy (Push Notification) và chuẩn bị hồ sơ ứng tuyển học bổng. Phân hệ này quản lý trạng thái dữ liệu thông qua cơ chế \textbf{Zustand} và \textbf{React Query} để đảm bảo tính đồng bộ tức thời. Hệ thống giao diện được thiết kế bằng thư viện UI \textbf{Tamagui}.
    \item \textbf{Web Admin Portal (React JS \& Vite):} Phân hệ web dành cho nhà tài trợ và ban quản trị trường học. Phân hệ này được xây dựng trên một repository độc lập sử dụng \textbf{React JS (v19)}, \textbf{Vite (v7)} và \textbf{Material UI}. Cung cấp các công cụ quản lý cơ sở dữ liệu học thuật quy mô lớn (Trường, Khoa, Môn), duyệt học bổng và hiển thị bảng điều khiển phát hiện gian lận ứng tuyển (Fraud Detection Dashboard).
    \item \textbf{Backend Monolith (NestJS):} Máy chủ trung tâm đóng vai trò xử lý toàn bộ logic nghiệp vụ (Business Logic), quản lý xác thực người dùng (Authentication), phân quyền dựa trên vai trò (Role-Based Access Control - RBAC), xử lý nghiệp vụ tài chính, quản lý hàng đợi BullMQ, và đóng vai trò proxy điều hướng an toàn yêu cầu tới AI Service. Backend sử dụng \textbf{TypeORM} để ánh xạ và tương tác với các thực thể cơ sở dữ liệu.
    \item \textbf{AI Service (FastAPI):} Máy chủ trí tuệ nhân tạo chuyên biệt chịu trách nhiệm chạy các LangChain Agent kết nối với Google Gemini API, thực hiện so khớp học bổng ngữ nghĩa (Semantic Matching) và phát hiện bất thường chi tiêu (Anomaly Detection). Các tác vụ AI chạy ngầm được xử lý bất đồng bộ thông qua \textbf{ARQ workers}.
\end{itemize}

\subsection{Luồng truyền thông điệp và tích hợp dịch vụ}
Các phân hệ trong hệ thống giao tiếp với nhau theo hai cơ chế chính:
\begin{enumerate}
    \item \textbf{Đồng bộ (Synchronous):} Sử dụng các RESTful API HTTP thông thường. Khi Mobile App hoặc Webadmin gửi yêu cầu cập nhật (như tạo giao dịch, sửa thông tin trường học), Backend sẽ ghi dữ liệu trực tiếp vào database PostgreSQL và trả về phản hồi lập tức.
    \item \textbf{Bất đồng bộ (Asynchronous):} Đối với các tác vụ tốn nhiều tài nguyên hoặc chạy ngầm (như quét phát hiện bất thường, gửi thông báo FCM, phân tích dữ liệu ứng tuyển), Backend sử dụng Redis làm Message Broker để đẩy các tác vụ vào hàng đợi BullMQ. Worker ngầm sẽ lấy tác vụ từ hàng đợi ra xử lý độc lập để tránh làm nghẽn tiến trình xử lý API chính của máy chủ.
\end{enumerate}

\subsection{Cơ chế xác thực giao tiếp giữa các dịch vụ (Service-to-Service Auth)}
Để đảm bảo an toàn thông tin và tránh việc kẻ xấu gọi trực tiếp vào lõi AI để bào mòn chi phí API Gemini, giao tiếp giữa NestJS Backend và FastAPI AI Service được bảo vệ nghiêm ngặt bằng cơ chế xác thực nội bộ:
\begin{itemize}
    \item API Gateway của Backend và FastAPI AI sử dụng chung một mã khóa bí mật được cấu hình qua biến môi trường \texttt{AI\_SERVICE\_SECRET}.
    \item Mỗi khi Backend gửi yêu cầu so khớp học bổng hoặc phân loại giao dịch tới FastAPI, nó sẽ đính kèm mã token này trong header của HTTP request dưới dạng Bearer Token. FastAPI AI Service sẽ giải mã và kiểm tra mã token này trước khi xử lý yêu cầu, đảm bảo chỉ có NestJS Backend được phép truy cập vào lõi AI.
\end{itemize}

\section{Thiết kế cơ sở dữ liệu (Database Design)}
Hệ thống sử dụng cơ sở dữ liệu quan hệ \textbf{PostgreSQL 15} \cite{postgresql-docs} làm nền tảng lưu trữ chính cho toàn bộ dữ liệu nghiệp vụ (giao dịch tài chính, hồ sơ học bổng, dữ liệu học thuật). Bên cạnh đó, \textbf{Redis 7} \cite{redis-docs} được sử dụng cho hai mục đích riêng biệt: làm Message Broker để quản lý hàng đợi BullMQ đẩy thông báo bất đồng bộ, và làm Cache lưu trữ kết quả phân tích AI Insights theo định dạng khóa \texttt{user\_id:month:year} với TTL 15 phút. Việc tính toán độ tương đồng trong thuật toán so khớp học bổng được thực hiện hoàn toàn trên bộ nhớ RAM tại AI Service (Python), không phụ thuộc vào các extension như \texttt{pgvector}.

\subsection{Sơ đồ thực thể liên kết (ERD) cho mô-đun Tài chính 6 lọ}
Các bảng dữ liệu phục vụ quản lý tài chính được thiết kế tối ưu nhằm lưu vết toàn bộ lịch sử chi dùng:
\begin{sloppypar}
\begin{itemize}
    \item \textbf{\texttt{money\_jars} (Các hũ tài chính):} Lưu trữ số dư và thông tin phân bổ của từng hũ.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh duy nhất cho hũ.
            \item \texttt{user\_id} (UUID, khóa ngoại): Tham chiếu đến bảng \texttt{users}.
            \item \texttt{jar\_code} (VARCHAR): Phân loại hũ (\texttt{nec}, \texttt{ffa}, \texttt{edu}, \texttt{ltss}, \texttt{play}, \texttt{give}).
            \item \texttt{name} (VARCHAR): Tên hiển thị của hũ.
            \item \texttt{percentage} (INT): Tỷ lệ phần trăm phân bổ (Tổng 6 lọ phải bằng 100\%).
            \item \texttt{balance} (DECIMAL): Số dư hiện tại của hũ.
        \end{itemize}
    \item \textbf{\texttt{financial\_transactions} (Giao dịch tài chính):} Lưu vết tất cả biến động dòng tiền.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh giao dịch.
            \item \texttt{user\_id} (UUID, khóa ngoại): Định danh người dùng thực hiện.
            \item \texttt{money\_jar\allowbreak\_id} (UUID, khóa ngoại): Tham chiếu đến hũ nguồn hoặc hũ đích.
            \item \texttt{amount} (DECIMAL): Số tiền thu hoặc chi.
            \item \texttt{type} (VARCHAR): Phân loại giao dịch (\texttt{income}, \texttt{expense}, \texttt{transfer}).
            \item \texttt{category} (VARCHAR): Danh mục chi tiết (ăn uống, đi lại, học tập...).
            \item \texttt{description} (TEXT): Mô tả chi tiết giao dịch.
            \item \texttt{transaction\allowbreak\_date} (TIMESTAMPTZ): Thời điểm giao dịch phát sinh.
        \end{itemize}
    \item \textbf{\texttt{budgets} (Hạn mức ngân sách):} Lưu trữ hạn mức chi tiêu tự cấu hình của người dùng.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh ngân sách.
            \item \texttt{user\_id} (UUID, khóa ngoại): Tham chiếu người dùng.
            \item \texttt{money\_jar\_id} (UUID, khóa ngoại): Lọ áp dụng hạn mức.
            \item \texttt{limit\_amount} (DECIMAL): Hạn mức chi tối đa cho phép.
            \item \texttt{spent\_amount} (DECIMAL): Số tiền thực tế đã chi trong chu kỳ.
            \item \texttt{month} (INT), \texttt{year} (INT): Khoảng thời gian áp dụng hạn mức.
        \end{itemize}
    \item \textbf{\texttt{auto\_transfer\_schedules} (Lịch trình chuyển tiền tự động):} Quản lý lịch trình tự động chuyển đổi số dư định kỳ.
    \item \textbf{\texttt{jar\_notification\_settings} (Cài đặt thông báo):} Quản lý cấu hình tùy chỉnh thông báo cho từng hũ tài chính.
    \item \textbf{\texttt{ai\_user\_preferences\_6jars} (Cấu hình AI):} Lưu vết cấu hình và sở thích của người dùng để AI phân loại giao dịch chính xác hơn.
\end{itemize}
\end{sloppypar}

\subsection{Sơ đồ thực thể liên kết (ERD) cho mô-đun Học bổng và Học thuật}
Các bảng dữ liệu nền tảng học thuật và học bổng bao gồm các liên kết khóa ngoại chặt chẽ, được chi tiết hóa các trường dữ liệu để phục vụ kiểm soát nghiệp vụ:
\begin{sloppypar}
\begin{itemize}
    \item \textbf{\texttt{universities} (Trường đại học):} Lưu trữ danh mục các trường liên kết. Gồm các cột:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh duy nhất của trường.
            \item \texttt{code} (VARCHAR): Mã viết tắt của trường (ví dụ: "HCMUS").
            \item \texttt{name} (VARCHAR): Tên đầy đủ của trường đại học.
            \item \texttt{city} (VARCHAR): Thành phố đặt trụ sở của trường.
            \item \texttt{status} (VARCHAR): Trạng thái hoạt động (\texttt{active}, \texttt{inactive}).
        \end{itemize}
    \item \textbf{\texttt{training\_programs} (Khoa/Viện đào tạo):} Lưu trữ thông tin ngành học, khoa và chương trình đào tạo của trường.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh duy nhất.
            \item \texttt{university\_id} (UUID, khóa ngoại): Tham chiếu đến trường đại học.
            \item \texttt{program\_code} (VARCHAR): Mã chương trình đào tạo.
            \item \texttt{program\_name} (VARCHAR): Tên khoa/chương trình.
        \end{itemize}
    \item \textbf{\texttt{university\_subjects} (Môn học):} Lưu trữ danh sách môn học của từng trường. Gồm các cột:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh duy nhất của môn học.
            \item \texttt{university\_id} (UUID, khóa ngoại): Tham chiếu đến trường đại học tương ứng.
            \item \texttt{subject\_code} (VARCHAR): Mã môn học (ví dụ: "COMP14002").
            \item \texttt{subject\_name} (VARCHAR): Tên môn học.
            \item \texttt{credits} (INT): Số tín chỉ của học phần.
            \item \texttt{status} (VARCHAR): Trạng thái môn học.
        \end{itemize}
    \item \textbf{\texttt{scholarship\_categories} (Danh mục học bổng):} Phân loại học bổng theo mục tiêu (VD: Học thuật, Tài năng, Hoàn cảnh khó khăn).
    \item \textbf{\texttt{scholarships} (Học bổng):} Lưu trữ thông tin chi tiết về các chương trình học bổng. Gồm các cột:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh học bổng.
            \item \texttt{name} (VARCHAR): Tên chương trình học bổng.
            \item \texttt{provider} (VARCHAR): Tên nhà tài trợ hoặc tổ chức cung cấp.
            \item \texttt{amount} (DECIMAL): Trị giá của mỗi suất học bổng (VNĐ).
            \item \texttt{quantity} (INT): Số lượng suất học bổng khả dụng.
            \item \texttt{application\_deadline} (DATE): Hạn cuối cùng nhận hồ sơ.
            \item \texttt{eligibility\_criteria} (TEXT): Văn bản mô tả yêu cầu điều kiện ứng tuyển.
            \item \texttt{is\_active} (BOOLEAN): Trạng thái xuất bản của học bổng.
        \end{itemize}
    \item \textbf{\texttt{scholarship\_requirements} (Điều kiện học bổng chi tiết):} Quy định các điều kiện mềm và điểm số. Gồm:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh yêu cầu.
            \item \texttt{scholarship\_id} (UUID, khóa ngoại): Tham chiếu đến học bổng.
            \item \texttt{title} (VARCHAR): Tiêu đề điều kiện (ví dụ: "Điểm trung bình học tập GPA").
            \item \texttt{description} (TEXT): Mô tả chi tiết yêu cầu (ví dụ: "GPA tối thiểu đạt 8.0").
            \item \texttt{is\_required} (BOOLEAN): Đánh dấu bắt buộc hoặc không bắt buộc.
        \end{itemize}
    \item \textbf{\texttt{scholarship\_documents} (Tài liệu yêu cầu ứng tuyển):} Các giấy tờ bắt buộc nộp kèm. Gồm:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh tài liệu.
            \item \texttt{scholarship\_id} (UUID, khóa ngoại): Tham chiếu đến học bổng.
            \item \texttt{document\_name} (VARCHAR): Tên loại tài liệu (ví dụ: "Bảng điểm học thuật", "Chứng chỉ IELTS").
            \item \texttt{document\_type} (VARCHAR): Định dạng tệp tin yêu cầu (\texttt{pdf}, \texttt{docx}, \texttt{png}).
            \item \texttt{is\_required} (BOOLEAN): Trạng thái bắt buộc.
        \end{itemize}
    \item \textbf{\texttt{student\_scholarships} (Hồ sơ ứng tuyển của sinh viên):} Quản lý tiến trình nộp hồ sơ (trước đây được thiết kế với tên \texttt{scholarship\_applications}). Gồm:
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh đơn nộp.
            \item \texttt{scholarship\_id} (UUID, khóa ngoại): Tham chiếu đến học bổng tương ứng.
            \item \texttt{user\_id} (UUID, khóa ngoại): Tham chiếu đến sinh viên ứng tuyển.
            \item \texttt{status} (VARCHAR): Trạng thái hồ sơ (\texttt{pending}, \texttt{under\_review}, \texttt{awarded}, \texttt{rejected}).
            \item \texttt{application\_date} (TIMESTAMPTZ): Thời điểm nộp đơn.
        \end{itemize}
    \item \textbf{\texttt{student\_scholarship\_documents} (Tài liệu đính kèm của sinh viên):} Lưu thông tin file minh chứng sinh viên đã tải lên.
    \item \textbf{\texttt{scholarship\_reviews} (Lịch sử đánh giá hồ sơ):} Quá trình xét duyệt hồ sơ của Ban quản trị.
    \item \textbf{\texttt{student\_grades} (Điểm số môn học):} Lưu điểm số môn học thực tế của sinh viên để làm minh chứng học bổng.
\end{itemize}
\end{sloppypar}

\section{Thiết kế chi tiết mô-đun Tài chính 6 lọ}
\subsection{Quy trình phân bổ nguồn thu và quản lý chi tiêu}
Quy trình phân bổ dòng tiền được thực hiện tự động nhằm giảm thiểu thao tác nhập liệu thủ công cho sinh viên:
\begin{enumerate}
    \item Khi người dùng nhận một khoản thu mới (ví dụ: Học bổng khuyến khích học tập trị giá $10.000.000$ VNĐ) và chọn hình thức phân bổ tự động.
    \item Hệ thống gọi API \texttt{POST /distribute-income}. Lõi Backend NestJS sẽ truy vấn cấu hình tỷ lệ phần trăm của 6 lọ hiện tại của người dùng (ví dụ cấu hình chuẩn: 55-10-10-10-10-5).
    \item Hệ thống tự động tính toán số tiền tương ứng cho từng lọ (ví dụ: hũ nhu cầu thiết yếu nhận $5.500.000$ VNĐ, hũ tự do tài chính nhận $1.000.000$ VNĐ...) và thực hiện ghi đồng thời (Batch write) 6 lệnh cộng tiền tương ứng vào cơ sở dữ liệu.
\end{enumerate}

\subsection{Cơ chế lập lịch giao dịch định kỳ (Auto-transfer \& Recurring)}
Để quản lý các khoản phí cố định phát sinh hàng tháng, hệ thống cho phép người dùng thiết lập lịch giao dịch:
\begin{itemize}
    \item Người dùng khai báo các khoản chi tiêu định kỳ (như Tiền nhà trọ, tiền mạng internet) kèm ngày thanh toán và hạn mức.
    \item Một Cron Job ngầm tại Backend chạy vào lúc 00:05 mỗi ngày để quét các lịch trình kích hoạt trong ngày hiện tại, tự động tạo giao dịch \texttt{expense} từ lọ được chỉ định và gửi thông báo nhắc nhở tới thiết bị của người dùng.
\end{itemize}

\subsection{Cơ chế đảm bảo tính toàn vẹn số dư hũ}
Nhằm ngăn ngừa hiện tượng sai lệch hoặc thất thoát số dư do lỗi bất đồng bộ mạng hoặc lỗi phần mềm, hệ thống áp dụng nguyên tắc:
\begin{itemize}
    \item \textbf{Số dư động (Dynamic Balance):} Số dư hiển thị trên giao diện của từng hũ không được lưu trữ tĩnh mà luôn được tính toán lại bằng công thức tích lũy từ lịch sử các dòng tiền thực tế:
    $$\text{Số dư hũ} = \sum (\text{Thu vào}) - \sum (\text{Chi ra}) \pm \sum (\text{Chuyển quỹ})$$
    \item \textbf{Hành vi khi xóa hũ tài chính:} Khi người dùng muốn xóa một hũ tài chính tùy chỉnh, hệ thống không chỉ đơn giản là xóa dòng dữ liệu trong DB. Nếu hũ đó có số dư khác 0, hệ thống bắt buộc người dùng chọn một hũ đích để nhận lại tiền. Sau đó, hệ thống thực thi một Transaction DB để tạo đồng thời một cặp giao dịch bù trừ: trừ tiền ở hũ nguồn (Giao dịch \texttt{expense}) và cộng tiền vào hũ đích (Giao dịch \texttt{income}) để đảm bảo tổng tài sản của người dùng hoàn toàn bất biến.
\end{itemize}

\subsection{Thuật toán phân loại giao dịch tự động bằng LLM (AI Transaction Classifier)}

\begin{itemize}
    \item Hệ thống gửi văn bản mô tả và số tiền tới LLM Classifier.
    \item Mô hình sử dụng Few-shot Prompting kết hợp cơ chế gán nhãn để phân tích ngữ nghĩa, tự động suy luận ra:
        \begin{itemize}
            \item "mua sach" $\rightarrow$ Lọ đề xuất: \texttt{education} (Học tập), Danh mục: \texttt{Sách vở/Học cụ}.
            \item "di cho" $\rightarrow$ Lọ đề xuất: \texttt{necessity} (Nhu cầu thiết yếu), Danh mục: \texttt{Thực phẩm/Ăn uống}.
        \end{itemize}
    \item \textbf{Cơ chế Override (Học hành vi người dùng):} Nếu AI phân loại sai hũ (ví dụ đề xuất lọ Hưởng thụ nhưng người dùng sửa thủ công thành lọ Cho đi trước khi lưu), hệ thống sẽ gọi API \texttt{/override} để lưu cặp dữ liệu sửa lỗi này vào bảng học tập của AI. Lần sau khi gặp từ khóa tương tự, AI sẽ ưu tiên lựa chọn sửa đổi của người dùng.
\end{itemize}

\subsection{Thuật toán so khớp học bổng thông minh dựa trên hồ sơ sinh viên}
Thuật toán so khớp học bổng được triển khai trong hàm \texttt{\_score\_profile\_match} tại AI Service, sử dụng tổ hợp có trọng số của ba độ đo tương đồng văn bản. Thuật toán hoạt động hoàn toàn trên bộ nhớ RAM của Python AI Service, không phụ thuộc vào cơ sở dữ liệu vector.

\textbf{Bước 1: Chuẩn hóa và gộp hồ sơ sinh viên.}
Hệ thống gộp toàn bộ các trường dữ liệu có giá trị (ngành học, trường, kỹ năng, hoạt động ngoại khóa) của sinh viên thành một chuỗi văn bản phẳng dùng hàm \texttt{\_build\_profile\_text}, sau đó áp dụng \textit{Unicode NFC normalization} để chuẩn hóa đồng nhất các ký tự tiếng Việt trên các hệ điều hành khác nhau.

\textbf{Bước 2: Tính độ tương đồng Jaccard trên tập token từ vựng (trọng số 0.5).}
Chỉ số Jaccard \cite{jaccard1901} được tính giữa tập hợp token từ vựng của hồ sơ $A$ và mô tả học bổng $B$:
\begin{equation}
\label{eq:jaccard}
J(A, B) = \frac{|A \cap B|}{|A \cup B|}
\end{equation}
Trong đó $A$ là tập các token thu được sau khi tokenize văn bản hồ sơ sinh viên, $B$ là tập token từ tiêu đề và mô tả học bổng.

\textbf{Bước 3: Tính độ tương đồng Jaccard trên Bigram ký tự (trọng số 0.3).}
Tập hợp n-gram ký tự độ dài $n=2$ (Bigrams) được định nghĩa theo \cite{cavnar1994-ngram}:
\begin{equation}
\label{eq:bigram}
N(S, 2) = \{s_i s_{i+1} \mid i = 1, 2, ..., |S|-1\}
\end{equation}
Ví dụ với chuỗi $S = \text{``cntt''} \rightarrow N(S,2) = \{\text{``cn''}, \text{``nt''}, \text{``tt''}\}$. Độ tương đồng Bigram giúp hệ thống nhận diện các từ viết tắt hoặc sai chính tả.

\textbf{Bước 4: Tính tỷ lệ khớp chuỗi SequenceMatcher (trọng số 0.2).}
Sử dụng mô-đun \texttt{difflib.SequenceMatcher} của Python để đo độ tương đồng của dãy ký tự giữa văn bản hồ sơ và tên học bổng.

\textbf{Bước 5: Kết hợp có trọng số.}
Điểm số khớp tổng hợp cuối cùng được tính theo công thức:
\begin{equation}
\label{eq:match-score}
\text{score} = 0.5 \times J_{token}(A, B) + 0.3 \times J_{bigram}(A, B) + 0.2 \times \text{Seq}(A, B)
\end{equation}

Giải thuật được mô tả chính thức như sau:

\begin{algorithm}
\caption{Thuật toán so khớp học bổng dựa trên hồ sơ sinh viên}
\label{alg:schol-match}
\begin{algorithmic}[1]
\Procedure{ScoreProfileMatch}{$Profile, Scholarship$}
    \State $profileText \gets \textsc{BuildProfileText}(Profile)$
    \State $profileText \gets \textsc{NfcNormalize}(profileText)$
    \State $scholText \gets Scholarship.name + \text{ } + Scholarship.eligibility\_criteria$
    \State $profileTokens \gets \textsc{Tokenize}(profileText)$
    \State $scholTokens \gets \textsc{Tokenize}(scholText)$
    \If{$profileTokens = \emptyset$ \textbf{or} $scholTokens = \emptyset$}
        \State \Return $0.0$
    \EndIf
    \State $tokenScore \gets J(profileTokens, scholTokens)$\Comment{Công thức~(\ref{eq:jaccard})}
    \State $profileBigrams \gets N(profileText, 2)$\Comment{Công thức~(\ref{eq:bigram})}
    \State $scholBigrams \gets N(scholText, 2)$
    \State $bigramScore \gets J(profileBigrams, scholBigrams)$
    \State $seqScore \gets \textsc{SequenceMatcher}(profileText,\, Scholarship.name)$
    \State $score \gets 0.5 \times tokenScore + 0.3 \times bigramScore + 0.2 \times seqScore$
    \State \Return $\textsc{Round}(score, 6)$
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Mô hình xử lý hàng đợi bất đồng bộ với Redis và BullMQ}

\end{enumerate}

\subsection{Cơ chế Idempotency và khử trùng lặp cảnh báo bất thường}
\begin{sloppypar}
Khi chạy công việc (job) quét phát hiện chi tiêu bất thường định kỳ vào cuối ngày, hệ thống có thể phát hiện cùng một hũ dành cho nhu cầu thiết yếu của người dùng liên tục bị vượt hạn mức hoặc tăng vọt đột biến. Để tránh làm phiền người dùng bằng việc gửi lặp đi lặp lại một thông báo cảnh báo:
\begin{itemize}
    \item Hệ thống áp dụng cơ chế loại bỏ trùng lặp (Deduplication) dựa trên khóa Idempotency.
    \item Khóa này được tính toán dựa trên tổ hợp: \texttt{user\_id + module\_type + alert\_type + target\_id + month}.
    \item Trước khi insert một cảnh báo bất thường mới vào bảng \texttt{ai\_anomaly\allowbreak\_alerts} và kích hoạt job gửi Push Notification, Worker sẽ kiểm tra trong database xem đã tồn tại cảnh báo nào có cùng khóa Idempotency trong tháng hiện tại chưa. Nếu đã có, hệ thống sẽ bỏ qua việc tạo mới và không gửi thêm thông báo đẩy.
\end{itemize}
\end{sloppypar}

\section{Thiết kế phân hệ Webadmin quản trị Master Data và Học bổng}
Phân hệ Webadmin được xây dựng trên nền tảng React JS, phục vụ công tác quản lý của Ban giám hiệu trường học và các Nhà tài trợ học bổng.

\subsection{Thiết kế quản lý phân cấp Trường học - Khoa - Môn học}
Dữ liệu học thuật được mô hình hóa theo cấu trúc cây phân cấp nghiêm ngặt nhằm phục vụ việc kiểm soát thông tin sinh viên:
\begin{itemize}
    \item Cấp 1: \textbf{Trường Đại học (University):} Định nghĩa mã trường, tên trường, khu vực địa lý.
    \item Cấp 2: \textbf{Khoa/Viện đào tạo (Faculty):} Liên kết trực tiếp trong hồ sơ học sinh, quy định ngành học chính của sinh viên.
    \item Cấp 3: \textbf{Môn học (Subject):} Quản lý mã môn học, tên môn học, số tín chỉ, thang điểm và trạng thái môn học.
\end{itemize}
Hệ thống sử dụng các ràng buộc khóa ngoại (Foreign Key Constraints) mức cơ sở dữ liệu để ngăn chặn việc xóa các thực thể cha (ví dụ xóa môn học khi môn học đó đã có điểm số của sinh viên), đảm bảo tính nhất quán của học bạ điện tử sinh viên.

\subsection{Quy trình kiểm soát, phê duyệt học bổng}
Quy trình được thiết kế khép kín nhằm tối đa hóa tính minh bạch:
\begin{enumerate}
    \item Nhà tài trợ đăng bài $\rightarrow$ Ban quản trị duyệt xuất bản $\rightarrow$ Sinh viên ứng tuyển trực tuyến.
    \item Webadmin cung cấp bảng điều khiển xét duyệt tập trung (Applications Review Portal) hiển thị chi tiết hồ sơ học thuật, file đính kèm của từng ứng viên.
    \item Admin có thể đối chiếu hồ sơ nộp vào với dữ liệu học thuật chuẩn trong Master Data (ngành học, điểm số môn học) qua giao diện trúc quan, giúp phát hiện chénh lệch dữ liệu mà không cần phải chạy thuật toán AI riêng.
\end{enumerate}

\begin{figure}[htbp]
