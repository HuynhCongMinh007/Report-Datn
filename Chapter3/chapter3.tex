\chapter{Thiết kế hệ thống}

\section{Kiến trúc tổng thể hệ thống (System Architecture)}
Hệ thống Student360 được thiết kế dựa trên mô hình monorepo đa dịch vụ nhằm đảm bảo tính phân tách trách nhiệm (Separation of Concerns), khả năng mở rộng độc lập và dễ dàng quản lý mã nguồn trong quá trình phát triển nhóm. Hệ thống bao gồm ba phân hệ chính hoạt động độc lập nhưng giao tiếp chặt chẽ với nhau thông qua mạng nội bộ và các giao thức kết nối tiêu chuẩn như HTTP REST, Server-Sent Events (SSE), và Message Queues.

\subsection{Kiến trúc Monorepo đa dịch vụ}
Cấu trúc hệ thống bao gồm bốn thành phần cốt lõi:
\begin{itemize}
    \item \textbf{Frontend Mobile Application (Expo - React Native):} Đây là ứng dụng di động đa nền tảng dành cho đối tượng sinh viên. Nó chịu trách nhiệm hiển thị giao diện người dùng, thu thập thông tin giao dịch tài chính, cung cấp cổng tương tác trò chuyện thời gian thực với trợ lý ảo, đăng ký nhận thông báo đẩy (Push Notification) và chuẩn bị hồ sơ ứng tuyển học bổng. Phân hệ này quản lý trạng thái dữ liệu thông qua cơ chế \textbf{Zustand} và \textbf{React Query} để đảm bảo tính đồng bộ tức thời. Hệ thống giao diện được thiết kế bằng thư viện UI \textbf{Tamagui}.
    \item \textbf{Web Admin Portal (React JS \& Vite):} Phân hệ web dành cho nhà tài trợ và ban quản trị trường học. Phân hệ này được xây dựng trên một repository độc lập sử dụng \textbf{React JS (v19)}, \textbf{Vite (v7)} và \textbf{Material UI}. Cung cấp các công cụ quản lý cơ sở dữ liệu học thuật quy mô lớn (trường đại học, khoa/chương trình đào tạo, môn học), dashboard nhà tài trợ, chức năng tạo học bổng, cấu hình điều kiện, quản lý tài liệu yêu cầu, xét duyệt hồ sơ ứng tuyển và hiển thị bảng điều khiển phát hiện gian lận ứng tuyển (Fraud Detection Dashboard).
    \item \textbf{Backend Monolith (NestJS):} Máy chủ trung tâm đóng vai trò xử lý toàn bộ logic nghiệp vụ (Business Logic), quản lý xác thực người dùng (Authentication), phân quyền dựa trên vai trò (Role-Based Access Control - RBAC), xử lý nghiệp vụ tài chính, quản lý hàng đợi BullMQ, và đóng vai trò proxy điều hướng an toàn yêu cầu tới AI Service. Backend sử dụng \textbf{TypeORM} để ánh xạ và tương tác với các thực thể cơ sở dữ liệu.
    \item \textbf{AI Service (FastAPI):} Máy chủ trí tuệ nhân tạo chuyên biệt chịu trách nhiệm chạy các LangChain Agent kết nối với Google Gemini API, thực hiện so khớp học bổng ngữ nghĩa (Semantic Matching) và phát hiện bất thường chi tiêu (Anomaly Detection). Các tác vụ AI chạy ngầm được xử lý bất đồng bộ thông qua \textbf{ARQ workers}.
\end{itemize}

\subsection{Luồng truyền thông điệp và tích hợp dịch vụ}
Các phân hệ trong hệ thống giao tiếp với nhau theo hai cơ chế chính:
\begin{enumerate}
    \item \textbf{Đồng bộ (Synchronous):} Sử dụng các RESTful API HTTP thông thường. Khi Mobile App hoặc Webadmin gửi yêu cầu cập nhật (như tạo giao dịch, sửa thông tin trường học), Backend sẽ ghi dữ liệu trực tiếp vào database PostgreSQL và trả về phản hồi lập tức.
    \item \textbf{Bất đồng bộ (Asynchronous):} Đối với các tác vụ tốn nhiều tài nguyên hoặc chạy ngầm (như quét phát hiện bất thường, gửi thông báo FCM, phân tích dữ liệu ứng tuyển), Backend sử dụng Redis làm Message Broker để đẩy các tác vụ vào hàng đợi BullMQ. Worker ngầm sẽ lấy tác vụ từ hàng đợi ra xử lý độc lập để tránh làm nghẽn tiến trình xử lý API chính của máy chủ.
\end{enumerate}

\subsection{Cơ chế xác thực giao tiếp giữa các dịch vụ (Service-to-Service Auth)}
Để đảm bảo an toàn thông tin và tránh việc kẻ xấu gọi trực tiếp vào lõi AI để bào mòn chi phí API Gemini, giao tiếp giữa NestJS Backend và FastAPI AI Service được bảo vệ nghiêm ngặt bằng cơ chế xác thực nội bộ:
\begin{itemize}
    \item API Gateway của Backend và FastAPI AI sử dụng chung một mã khóa bí mật được cấu hình qua biến môi trường \texttt{AI\_SERVICE\_SECRET}.
    \item Mỗi khi Backend gửi yêu cầu so khớp học bổng hoặc phân loại giao dịch tới FastAPI, nó sẽ đính kèm mã token này trong header của HTTP request dưới dạng Bearer Token. FastAPI AI Service sẽ giải mã và kiểm tra mã token này trước khi xử lý yêu cầu, đảm bảo chỉ có NestJS Backend được phép truy cập vào lõi AI.
\end{itemize}

\section{Thiết kế cơ sở dữ liệu (Database Design)}
Hệ thống sử dụng cơ sở dữ liệu quan hệ \textbf{PostgreSQL 15} \cite{postgresql-docs} làm nền tảng lưu trữ chính cho toàn bộ dữ liệu nghiệp vụ (giao dịch tài chính, hồ sơ học bổng, dữ liệu học thuật). Bên cạnh đó, \textbf{Redis 7} \cite{redis-docs} được sử dụng chủ yếu làm Message Broker để quản lý hàng đợi BullMQ đẩy thông báo bất đồng bộ và các job lên lịch giao dịch định kỳ. Riêng kết quả phân tích AI Insights không lưu qua Redis mà được lưu bền vững trong bảng PostgreSQL \texttt{ai\_finance\_insight\_snapshot}, theo khóa \texttt{user\_id + tháng + năm + ngày phân tích}, giúp các lượt truy cập báo cáo trong cùng chu kỳ không cần gọi lại LLM. Việc tính toán độ tương đồng trong cơ chế gợi ý học bổng được thực hiện hoàn toàn trên bộ nhớ RAM tại AI Service (Python), không phụ thuộc vào các extension như \texttt{pgvector}.

\subsection{Sơ đồ thực thể liên kết (ERD) cho mô-đun Tài chính 6 lọ}
Các bảng dữ liệu phục vụ quản lý tài chính được thiết kế tối ưu nhằm lưu vết toàn bộ lịch sử chi dùng:
\begin{sloppypar}
\begin{itemize}
    \item \textbf{\texttt{money\_jars} (Các hũ tài chính):} Lưu trữ số dư và thông tin phân bổ của từng hũ.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh duy nhất cho hũ.
            \item \texttt{user\_id} (UUID, khóa ngoại): Tham chiếu đến bảng \texttt{users}.
            \item \texttt{category\_type} (VARCHAR): Phân loại hũ, gồm 6 giá trị chuẩn \texttt{essentials} (Nhu cầu thiết yếu), \texttt{education} (Giáo dục), \texttt{investment} (Tự do tài chính), \texttt{reserve} (Tiết kiệm dài hạn), \texttt{enjoyment} (Hưởng thụ) và \texttt{sharing} (Cho đi).
            \item \texttt{name} (VARCHAR): Tên hiển thị của hũ.
            \item \texttt{percentage} (DECIMAL(5,2)): Tỷ lệ phần trăm phân bổ (Tổng 6 lọ phải bằng 100\%).
            \item \texttt{current\_balance} (DECIMAL(15,2)): Số dư hiện tại của hũ, được đồng bộ lại định kỳ từ tổng hợp lịch sử giao dịch (không phải giá trị tính hoàn toàn tại thời điểm đọc).
            \item \texttt{accumulated\_income}, \texttt{accumulated\_expense} (DECIMAL): Tổng thu và tổng chi lũy kế, dùng làm dữ liệu nguồn cho việc tính lại \texttt{current\_balance}.
        \end{itemize}
    \item \textbf{\texttt{financial\_transactions} (Giao dịch tài chính):} Lưu vết tất cả biến động dòng tiền.
        \begin{itemize}
            \item \texttt{id} (UUID, khóa chính): Định danh giao dịch.
            \item \texttt{user\_id} (UUID, khóa ngoại, qua quan hệ): Định danh người dùng thực hiện.
            \item \texttt{money\_jar\allowbreak\_id} (UUID, khóa ngoại): Tham chiếu đến hũ nguồn.
            \item \texttt{counterpart\_jar\_id} (UUID, khóa ngoại, có thể NULL): Hũ đích khi giao dịch là một lượt chuyển quỹ giữa 2 hũ; hệ thống không dùng riêng một giá trị \texttt{type = transfer} mà mô hình hóa chuyển quỹ bằng cặp bản ghi thu/chi có \texttt{counterpart\_jar\_id} trỏ lẫn nhau.
            \item \texttt{amount} (DECIMAL): Số tiền thu hoặc chi.
            \item \texttt{type} (ENUM): Chỉ gồm 2 giá trị \texttt{income} và \texttt{expense}.
            \item \texttt{tags} (TEXT[]): Danh sách nhãn phân loại chi tiết gắn với giao dịch (thay cho một cột \texttt{category} đơn), tham chiếu tới các bản ghi trong bảng \texttt{budgets} (xem bên dưới).
            \item \texttt{notes} (TEXT): Mô tả chi tiết giao dịch.
            \item \texttt{transaction\allowbreak\_date} (TIMESTAMPTZ): Thời điểm giao dịch phát sinh.
        \end{itemize}
    \item \textbf{\texttt{budgets} (Nhãn chi tiêu theo hũ -- Jar Tags):} Về mặt tên bảng và cấu trúc cột, đây là bảng được thiết kế ban đầu cho hạn mức ngân sách, nhưng hiện tại được hệ thống tái sử dụng làm cơ chế gắn nhãn/danh mục con (tag) cho từng hũ (ví dụ hũ Nhu cầu thiết yếu có các tag "Ăn uống", "Đi lại", "Hóa đơn"...), phục vụ việc phân loại chi tiêu chi tiết hơn 6 hũ gốc. Các trường chính: \texttt{id}, \texttt{money\_jar\_id} (hũ sở hữu tag), \texttt{name}, \texttt{slug}, \texttt{is\_tag} (luôn \texttt{true} trong luồng hiện tại), \texttt{is\_default}, \texttt{amount}/\texttt{spent\_amount}/\texttt{remaining\_amount} (giữ lại từ thiết kế hạn mức cũ nhưng không được cập nhật qua nghiệp vụ đang hoạt động), \texttt{period\_type} (\texttt{monthly}, \texttt{yearly}, \texttt{one\_time}).
    \item \textbf{\texttt{auto\_transfer\_schedules} (Lịch trình chuyển tiền tự động):} Quản lý lịch trình tự động chuyển đổi số dư định kỳ.
    \item \textbf{\texttt{jar\_notification\_settings} (Ngưỡng cảnh báo chi tiêu theo hũ):} Đây mới là cơ chế thực sự đóng vai trò "hạn mức chi tiêu" đang hoạt động trong hệ thống. Mỗi hũ có tối đa một bản ghi cấu hình ngưỡng cảnh báo, gồm \texttt{money\_jar\_id} (khóa ngoại, ràng buộc duy nhất 1-1 với hũ), \texttt{percent\_enabled}/\texttt{percent\_value} (bật cảnh báo khi chi tiêu vượt X\% số dư đầu kỳ) và \texttt{amount\_enabled}/\texttt{amount\_value} (bật cảnh báo khi chi tiêu vượt một số tiền cố định). Cấu hình này không giới hạn theo tháng/năm cụ thể mà áp dụng liên tục cho tới khi người dùng thay đổi.
    \item \textbf{\texttt{ai\_user\_preferences\_6jars} (Cấu hình AI):} Lưu vết cấu hình và sở thích của người dùng để AI phân loại giao dịch chính xác hơn.
\end{itemize}
\end{sloppypar}

\subsection{Sơ đồ thực thể liên kết (ERD) cho mô-đun Học bổng và Học thuật}
Các bảng dữ liệu nền tảng học thuật và học bổng bao gồm các liên kết khóa ngoại chặt chẽ, được chi tiết hóa các trường dữ liệu để phục vụ kiểm soát nghiệp vụ:
\begin{sloppypar}
\begin{itemize}
    \item \textbf{\texttt{universities} (Trường đại học):} Lưu danh mục trường dùng cho Webadmin và Mobile App. Các trường chính gồm \texttt{id}, \texttt{name}, \texttt{code}, \texttt{type}, \texttt{city}, \texttt{status}, \texttt{created\_at} và \texttt{updated\_at}.
    \item \textbf{\texttt{training\_programs} (Khoa/Chương trình đào tạo):} Lưu danh mục khoa hoặc chương trình đào tạo thuộc từng trường. Các trường chính gồm \texttt{id}, \texttt{university\_id}, \texttt{program\_code}, \texttt{program\_name}, \texttt{description}, \texttt{status}, \texttt{created\_at} và \texttt{updated\_at}.
    \item \textbf{\texttt{university\_subjects} (Môn học):} Lưu danh mục môn học theo từng trường. Các trường chính gồm \texttt{id}, \texttt{university\_id}, \texttt{subject\_code}, \texttt{subject\_name}, \texttt{course\_code}, \texttt{credits}, \texttt{grade\_scale}, \texttt{description}, \texttt{status}, \texttt{created\_at} và \texttt{updated\_at}.
    \item \textbf{\texttt{student\_academic\_profiles} (Hồ sơ học thuật sinh viên):} Lưu thông tin học thuật của sinh viên, gồm \texttt{user\_id}, \texttt{university}, \texttt{student\_code}, \texttt{education\_program}, \texttt{degree\_level}, \texttt{faculty}, \texttt{major}, \texttt{minor}, \texttt{program\_type}, \texttt{gpa}, \texttt{enrollment\_year}, \texttt{expected\_graduation\_year}, \texttt{current\_status}, \texttt{current\_year}, \texttt{current\_semester}. Mục tiêu tín chỉ cá nhân không tách thành bảng riêng mà được nhúng trực tiếp vào bảng này qua 2 cột bổ sung \texttt{target\_credits} và \texttt{target\_note} (được thêm bằng migration riêng), phục vụ dashboard tiến độ tín chỉ.
    \item \textbf{\texttt{student\_grades} (Bảng điểm sinh viên):} Lưu điểm từng môn do sinh viên nhập hoặc import. Dữ liệu gắn với sinh viên và môn học, gồm \texttt{subject\_id}, \texttt{grade} và \texttt{student\_note}. Hệ thống sử dụng thông tin \texttt{credits} và \texttt{grade\_scale} từ \texttt{university\_subjects} để kiểm tra điểm hợp lệ và tính GPA.
    \item \textbf{\texttt{scholarship\_categories} (Danh mục học bổng):} Phân loại các chương trình học bổng theo mục tiêu (ví dụ: Học thuật, Tài năng, Vượt khó). Các trường chính gồm \texttt{id}, \texttt{name}, \texttt{description}, \texttt{created\_at} và \texttt{updated\_at}.
    \item \textbf{\texttt{scholarships} (Học bổng):} Lưu trữ thông tin chương trình học bổng do nhà tài trợ công bố. Các trường chính gồm \texttt{id}, \texttt{category\_id} (khóa ngoại trỏ đến danh mục học bổng), \texttt{name}, \texttt{description}, \texttt{provider}, \texttt{provider\_id}, \texttt{amount}, \texttt{currency}, \texttt{quantity}, \texttt{minimum\_gpa}, \texttt{minimum\_gpa\_scale}, \texttt{target\_majors}, \texttt{target\_universities}, \texttt{application\_deadline}, \texttt{result\_announcement\_date}, \texttt{benefits}, \texttt{contact\_email}, \texttt{contact\_phone}, \texttt{official\_website} và \texttt{is\_active}.
    \item \textbf{\texttt{scholarship\_requirements} (Điều kiện học bổng):} Lưu các điều kiện chi tiết do nhà tài trợ cấu hình cho từng học bổng, gồm \texttt{title}, \texttt{description}, \texttt{is\_required} và \texttt{sort\_order}.
    \item \textbf{\texttt{scholarship\_documents} (Tài liệu yêu cầu):} Định nghĩa các loại tài liệu sinh viên cần nộp, gồm \texttt{document\_name}, \texttt{document\_type}, \texttt{is\_required}, \texttt{max\_file\_size\_mb} và \texttt{sample\_url}.
    \item \textbf{\texttt{student\_scholarships} (Hồ sơ ứng tuyển):} Lưu hồ sơ ứng tuyển của sinh viên cho từng học bổng. Bảng có ràng buộc duy nhất trên cặp \texttt{user\_id} và \texttt{scholarship\_id} để ngăn nộp trùng. Trạng thái hồ sơ gồm \texttt{draft}, \texttt{submitted}, \texttt{under\_review}, \texttt{need\_modification}, \texttt{approved}, \texttt{rejected}, \texttt{awarded} và \texttt{cancelled}.
    \item \textbf{\texttt{student\_scholarship\_documents} (Tài liệu hồ sơ):} Lưu các file sinh viên đã upload cho từng loại tài liệu yêu cầu, gồm \texttt{file\_url}, \texttt{upload\_date}, \texttt{status} và \texttt{reviewer\_note}.
    \item \textbf{\texttt{student\_scholarship\_history} (Lịch sử trạng thái hồ sơ):} Ghi nhận các lần chuyển trạng thái của hồ sơ, gồm \texttt{from\_status}, \texttt{to\_status}, \texttt{note}, \texttt{changed\_by} và \texttt{created\_at}.
    \item \textbf{\texttt{scholarship\_reviews} (Quá trình đánh giá hồ sơ):} Lưu vết quá trình xét duyệt hồ sơ của Ban quản lý/Nhà tài trợ. Các trường chính gồm \texttt{id}, \texttt{student\_scholarship\_id}, \texttt{reviewer\_id}, \texttt{stage}, \texttt{status}, \texttt{comment}, \texttt{reviewed\_at}, \texttt{created\_at} và \texttt{updated\_at}.
\end{itemize}
\end{sloppypar}

\section{Thiết kế chi tiết mô-đun Tài chính 6 lọ}
\subsection{Quy trình phân bổ nguồn thu và quản lý chi tiêu}
Quy trình phân bổ dòng tiền được thực hiện tự động nhằm giảm thiểu thao tác nhập liệu thủ công cho sinh viên:
\begin{enumerate}
    \item Khi người dùng nhận một khoản thu mới (ví dụ: Học bổng khuyến khích học tập trị giá $10.000.000$ VNĐ) và chọn hình thức phân bổ tự động.
    \item Hệ thống gọi API \texttt{POST /finance/transactions/distribute-income}. Lõi Backend NestJS sẽ truy vấn cấu hình tỷ lệ phần trăm của 6 lọ hiện tại của người dùng (ví dụ cấu hình chuẩn: 55-10-10-10-10-5).
    \item Hệ thống tự động tính toán số tiền tương ứng cho từng lọ (ví dụ: hũ nhu cầu thiết yếu nhận $5.500.000$ VNĐ, hũ tự do tài chính nhận $1.000.000$ VNĐ...) và thực hiện ghi đồng thời (Batch write) 6 lệnh cộng tiền tương ứng vào cơ sở dữ liệu.
\end{enumerate}

\subsection{Cơ chế lập lịch giao dịch định kỳ (Auto-transfer \& Recurring)}
Để quản lý các khoản phí cố định phát sinh hàng tháng, hệ thống cho phép người dùng thiết lập lịch giao dịch:
\begin{itemize}
    \item Người dùng khai báo các khoản chi tiêu định kỳ (như Tiền nhà trọ, tiền mạng internet) kèm ngày thanh toán và hũ áp dụng.
    \item Khác với cơ chế Cron Job quét toàn bộ theo chu kỳ cố định, hệ thống lập lịch theo hướng sự kiện: mỗi khi một lịch trình (\texttt{AutoTransferSchedule}) được tạo hoặc cập nhật, \texttt{AutoTransferQueueService} tại Backend tính thời điểm chạy kế tiếp (\texttt{nextRunDate}) và đẩy một BullMQ Delayed Job riêng cho đúng lịch trình đó (\texttt{delay} bằng khoảng thời gian tới \texttt{nextRunDate}). Khi Backend khởi động lại, toàn bộ lịch trình đang hoạt động được đối soát và lên lại lịch một lần (\texttt{onApplicationBootstrap}) để đảm bảo không bị mất job do restart. Khi job được kích hoạt đúng thời điểm, Worker tự động tạo giao dịch \texttt{expense} từ hũ được chỉ định và gửi thông báo nhắc nhở tới thiết bị của người dùng.
\end{itemize}

\subsection{Cơ chế đảm bảo tính toàn vẹn số dư hũ}
Nhằm ngăn ngừa hiện tượng sai lệch hoặc thất thoát số dư do lỗi bất đồng bộ mạng hoặc lỗi phần mềm, hệ thống áp dụng nguyên tắc:
\begin{itemize}
    \item \textbf{Số dư động (Dynamic Balance):} Số dư hiển thị trên giao diện của từng hũ không được lưu trữ tĩnh mà luôn được tính toán lại bằng công thức tích lũy từ lịch sử các dòng tiền thực tế:
    $$\text{Số dư hũ} = \sum (\text{Thu vào}) - \sum (\text{Chi ra}) \pm \sum (\text{Chuyển quỹ})$$
    \item \textbf{Hành vi khi xóa hũ tài chính:} Khi người dùng muốn xóa một hũ tài chính tùy chỉnh, hệ thống không chỉ đơn giản là xóa dòng dữ liệu trong DB. Nếu hũ đó có số dư khác 0, hệ thống bắt buộc người dùng chọn một hũ đích để nhận lại tiền. Sau đó, hệ thống tạo đồng thời một cặp giao dịch bù trừ: trừ tiền ở hũ nguồn (Giao dịch \texttt{expense}) và cộng tiền vào hũ đích (Giao dịch \texttt{income}) để đảm bảo tổng tài sản của người dùng không đổi, rồi tính lại số dư của cả hai hũ liên quan. Hiện tại thao tác này được thực hiện bằng một lệnh ghi đồng thời cả hai bản ghi giao dịch, chưa được bọc trong một \texttt{QueryRunner} Transaction tường minh như cơ chế chuyển khoản thông thường -- đây là một điểm cần bổ sung để đảm bảo an toàn tuyệt đối trong trường hợp lỗi xảy ra giữa chừng.
\end{itemize}

\subsection{Cơ chế gợi ý học bổng thông minh dựa trên hồ sơ sinh viên}
Cơ chế gợi ý học bổng được triển khai trong hàm \texttt{\_score\_profile\_match} tại AI Service, sử dụng tổ hợp có trọng số của ba độ đo tương đồng văn bản. Thuật toán hoạt động hoàn toàn trên bộ nhớ RAM của Python AI Service, không phụ thuộc vào cơ sở dữ liệu vector.

\textbf{Bước 1: Chuẩn hóa và gộp hồ sơ sinh viên.}
Hệ thống gộp toàn bộ các trường dữ liệu có giá trị (ngành học, trường, kỹ năng, hoạt động ngoại khóa) của sinh viên thành một chuỗi văn bản phẳng dùng hàm \texttt{\_build\_profile\_text}, sau đó áp dụng \textit{Unicode NFC normalization} để chuẩn hóa đồng nhất các ký tự tiếng Việt trên các hệ điều hành khác nhau.

\textbf{Bước 2: Tính độ tương đồng Jaccard trên tập token từ vựng (trọng số 0.5).}
Chỉ số Jaccard \cite{jaccard1901} được tính giữa tập hợp token từ vựng của hồ sơ $A$ và mô tả học bổng $B$:
\begin{equation}
\label{eq:jaccard}
J(A, B) = \frac{|A \cap B|}{|A \cup B|}
\end{equation}
Trong đó $A$ là tập các token thu được sau khi tokenize văn bản hồ sơ sinh viên, $B$ là tập token từ tiêu đề và mô tả học bổng.

\textbf{Bước 3: Tính độ tương đồng Jaccard trên Bigram ký tự (trọng số 0.3).}
Tập hợp n-gram ký tự độ dài $n=2$ (Bigrams) được định nghĩa theo \cite{cavnar1994-ngram}:
\begin{equation}
\label{eq:bigram}
N(S, 2) = \{s_i s_{i+1} \mid i = 1, 2, ..., |S|-1\}
\end{equation}
Ví dụ với chuỗi $S = \text{``cntt''} \rightarrow N(S,2) = \{\text{``cn''}, \text{``nt''}, \text{``tt''}\}$. Độ tương đồng Bigram giúp hệ thống nhận diện các từ viết tắt hoặc sai chính tả.

\textbf{Bước 4: Tính tỷ lệ khớp chuỗi SequenceMatcher (trọng số 0.2).}
Sử dụng mô-đun \texttt{difflib.SequenceMatcher} của Python để đo độ tương đồng của dãy ký tự giữa văn bản hồ sơ và tên học bổng.

\textbf{Bước 5: Kết hợp có trọng số.}
Điểm số khớp tổng hợp cuối cùng được tính theo công thức:
\begin{equation}
\label{eq:match-score}
\text{score} = 0.5 \times J_{token}(A, B) + 0.3 \times J_{bigram}(A, B) + 0.2 \times \text{Seq}(A, B)
\end{equation}

Giải thuật được mô tả chính thức như sau:

\begin{algorithm}
\caption{Cơ chế gợi ý học bổng dựa trên hồ sơ sinh viên}
\label{alg:schol-match}
\begin{algorithmic}[1]
\Procedure{ScoreProfileMatch}{$Profile, Scholarship$}
    \State $profileText \gets \textsc{BuildProfileText}(Profile)$
    \State $profileText \gets \textsc{NfcNormalize}(profileText)$
    \State $scholText \gets Scholarship.name + \text{ } + Scholarship.eligibility\_criteria$
    \State $profileTokens \gets \textsc{Tokenize}(profileText)$
    \State $scholTokens \gets \textsc{Tokenize}(scholText)$
    \If{$profileTokens = \emptyset$ \textbf{or} $scholTokens = \emptyset$}
        \State \Return $0.0$
    \EndIf
    \State $tokenScore \gets J(profileTokens, scholTokens)$\Comment{Công thức~(\ref{eq:jaccard})}
    \State $profileBigrams \gets N(profileText, 2)$\Comment{Công thức~(\ref{eq:bigram})}
    \State $scholBigrams \gets N(scholText, 2)$
    \State $bigramScore \gets J(profileBigrams, scholBigrams)$
    \State $seqScore \gets \textsc{SequenceMatcher}(profileText,\, Scholarship.name)$
    \State $score \gets 0.5 \times tokenScore + 0.3 \times bigramScore + 0.2 \times seqScore$
    \State \Return $\textsc{Round}(score, 6)$
\EndProcedure
\end{algorithmic}
\end{algorithm}

\subsection{Cơ chế lưu trữ snapshot nhận định tài chính (AI Insights) tối ưu năng lực đáp ứng}
Để giảm thiểu số lượng cuộc gọi trực tiếp đến Mô hình ngôn ngữ lớn (LLM) và tối ưu hóa thời gian phản hồi, hệ thống không dùng bộ nhớ đệm tạm thời (cache) cho kết quả phân tích AI Insights mà lưu trực tiếp thành snapshot bền vững trong PostgreSQL.
\begin{itemize}
    \item \textbf{Mục đích sử dụng:} Khi người dùng truy cập màn hình báo cáo, hệ thống cần cung cấp các nhận định tài chính tức thời. Do việc gọi API LLM tốn thời gian và chi phí tài nguyên, dữ liệu nhận định sau khi sinh ra lần đầu trong một chu kỳ báo cáo sẽ được lưu lại thay vì phải sinh lại từ đầu.
    \item \textbf{Cơ chế hoạt động:} Kết quả được ghi vào bảng \texttt{ai\_finance\_insight\_snapshot} với khóa xác định duy nhất theo \texttt{user\_id + tháng + năm + ngày phân tích}. Các yêu cầu truy cập tiếp theo trong cùng chu kỳ sẽ được đáp ứng trực tiếp từ bản ghi snapshot này mà không cần kích hoạt lại quy trình phân tích của LLM, giúp giảm thiểu độ trễ và tiết kiệm chi phí vận hành API. Để tránh trường hợp một job sinh insight bị kẹt ở trạng thái "đang xử lý" (\texttt{GENERATING}) do lỗi giữa chừng, hệ thống áp dụng thêm một cửa sổ hết hạn 5 phút (\texttt{staleAt}), cho phép job mới được kích hoạt lại thay vì chờ vô thời hạn.
\end{itemize}

\subsection{Cơ chế Idempotency và khử trùng lặp cảnh báo bất thường}
\begin{sloppypar}
Khi chạy công việc (job) quét phát hiện chi tiêu bất thường định kỳ vào cuối ngày (Cron Job chạy lúc 23:00 hằng ngày tại Backend NestJS), hệ thống có thể phát hiện cùng một giao dịch bị đánh giá bất thường nhiều lần nếu job chạy lại hoặc xử lý trùng. Để tránh gửi lặp một cảnh báo cho cùng một giao dịch:
\begin{itemize}
    \item Hệ thống áp dụng cơ chế loại bỏ trùng lặp (Deduplication) dựa trên khóa Idempotency.
    \item Khóa này được tính toán dựa trên tổ hợp: \texttt{user\_id + target\_id} (\texttt{target\_id} là mã giao dịch tài chính đang được đánh giá).
    \item Trước khi insert một cảnh báo bất thường mới vào bảng \texttt{ai\_anomaly\allowbreak\_alerts} và kích hoạt job gửi Push Notification, Worker sẽ kiểm tra trong database xem đã tồn tại cảnh báo nào có cùng người dùng và cùng giao dịch nguồn chưa. Nếu đã có, hệ thống sẽ bỏ qua việc tạo mới và không gửi thêm thông báo đẩy.
    \item \textit{Giới hạn hiện tại:} vì khóa chỉ dựa trên từng giao dịch riêng lẻ, cơ chế này chưa xử lý được kịch bản "cùng một hũ liên tục vượt hạn mức trong tháng" bằng một cảnh báo gộp duy nhất như định hướng thiết kế ban đầu -- mỗi giao dịch bất thường vẫn có thể sinh ra một cảnh báo (và một thông báo đẩy) riêng.
\end{itemize}
\end{sloppypar}

\section{Thiết kế phân hệ Webadmin quản trị Master Data và Học bổng}
Phân hệ Webadmin được xây dựng trên nền tảng React JS, phục vụ công tác quản lý của Ban giám hiệu trường học và các Nhà tài trợ học bổng.

\subsection{Quy trình quản lý dữ liệu học thuật và bảng điểm}
Quy trình được thiết kế theo hai nhóm người dùng chính: quản trị viên và sinh viên.

\begin{enumerate}
    \item \textbf{Quản trị viên quản lý trường:} Webadmin lấy danh sách trường qua \texttt{GET /admin/universities}, tạo mới qua \texttt{POST /admin/universities}, cập nhật qua \texttt{PUT /admin/universities/\{id\}} và xóa qua \texttt{DELETE /admin/universities/\{id\}}.
    \item \textbf{Quản trị viên quản lý khoa/chương trình đào tạo:} Với từng trường, Webadmin lấy danh sách khoa/chương trình đào tạo qua \texttt{GET /admin/universities/\{id\}/training-programs}, thêm mới qua \texttt{POST /admin/universities/\{id\}/training-programs}, cập nhật qua \texttt{PUT /admin/training-programs/\{id\}}, xóa qua \texttt{DELETE /admin/training-programs/\{id\}} và import hàng loạt qua \texttt{POST /admin/universities/\{id\}/training-programs/bulk}.
    \item \textbf{Quản trị viên quản lý môn học:} Với từng trường, Webadmin lấy danh sách môn qua \texttt{GET /admin/universities/\{id\}/subjects}, thêm mới qua \texttt{POST /admin/universities/\{id\}/subjects}, cập nhật qua \texttt{PUT /admin/subjects/\{id\}}, xóa qua \texttt{DELETE /admin/subjects/\{id\}} và import hàng loạt qua \texttt{POST /admin/universities/\{id\}/subjects/bulk}.
    \item \textbf{Sinh viên chọn thông tin học thuật:} Mobile App lấy danh sách trường qua \texttt{GET /universities/dropdown} và danh sách khoa/chương trình đào tạo theo trường qua \texttt{GET /universities/\{universityId\}/training-programs/dropdown}. Thông tin này được dùng trong hồ sơ đăng ký và hồ sơ cá nhân.
    \item \textbf{Sinh viên xem danh mục môn học:} Mobile App lấy danh mục môn của trường qua \texttt{GET /student/university-subjects/dropdown}. Mỗi môn có mã môn, mã học phần, số tín chỉ và thang điểm để phục vụ nhập điểm.
    \item \textbf{Sinh viên nhập hoặc import điểm:} Sinh viên thêm hoặc cập nhật điểm qua \texttt{POST /student/grades}. File TXT/CSV được parse ở Mobile App, map môn theo \texttt{subject\_id}, \texttt{subject\_code} hoặc \texttt{course\_code}, cho phép chỉnh sửa trước khi lưu.
    \item \textbf{Sinh viên quản lý bảng điểm:} Sinh viên xem danh sách điểm qua \texttt{GET /student/grades}, xóa điểm qua \texttt{DELETE /student/grades/\{gradeId\}}, xem dashboard qua \texttt{GET /student/dashboard} và cập nhật mục tiêu tín chỉ qua \texttt{PATCH /student/gradebook-target}.
\end{enumerate}

\subsection{Quy trình tạo, nộp và xét duyệt hồ sơ học bổng}
Quy trình học bổng được thiết kế theo hai vai trò chính: nhà tài trợ và sinh viên. Quy trình tổng quan gồm các bước:
\begin{enumerate}
    \item Nhà tài trợ đăng bài $\rightarrow$ Ban quản trị duyệt xuất bản $\rightarrow$ Sinh viên ứng tuyển trực tuyến.
    \item Webadmin cung cấp bảng điều khiển xét duyệt tập trung (Applications Review Portal) hiển thị chi tiết hồ sơ học thuật, file đính kèm của từng ứng viên.
    \item Admin có thể đối chiếu hồ sơ nộp vào với dữ liệu học thuật chuẩn trong Master Data (ngành học, điểm số môn học) qua giao diện trực quan, giúp phát hiện chênh lệch dữ liệu mà không cần phải chạy thuật toán AI riêng.
\end{enumerate}

Chi tiết các bước thực hiện ở mức hệ thống và API được mô tả như sau:
\begin{enumerate}
    \item \textbf{Nhà tài trợ tạo học bổng:} Nhà tài trợ đăng nhập Webadmin, nhập thông tin chương trình học bổng và gửi yêu cầu \texttt{POST /scholarships}. Backend lưu bản ghi vào bảng \texttt{scholarships}, trong đó \texttt{provider\_id} liên kết học bổng với tổ chức tài trợ trong bảng \texttt{organizations}.
    \item \textbf{Cấu hình điều kiện và tài liệu:} Sau khi tạo học bổng, nhà tài trợ cấu hình các điều kiện xét tuyển qua \texttt{POST /scholarships/\{id\}/requirements} và danh sách tài liệu yêu cầu qua \texttt{POST /scholarships/\{id\}/documents}.
    \item \textbf{Sinh viên xem và chuẩn bị hồ sơ:} Ứng dụng di động lấy danh sách học bổng qua \texttt{GET /scholarships} và chi tiết từng học bổng qua \texttt{GET /scholarships/\{id\}}. Sinh viên nhập CCCD, GPA, ghi chú và upload các tài liệu minh chứng theo danh sách yêu cầu.
    \item \textbf{Lưu nháp hoặc nộp hồ sơ:} Sinh viên gửi hồ sơ qua \texttt{POST /scholarships/register}. Nếu chọn lưu nháp, hệ thống tạo hồ sơ ở trạng thái \texttt{draft}; nếu nộp chính thức, hồ sơ chuyển sang \texttt{submitted}.
    \item \textbf{Kiểm tra hợp lệ ở Backend:} Backend kiểm tra hạn nộp, tình trạng hồ sơ cá nhân, GPA tối thiểu, định dạng CCCD và ràng buộc không nộp trùng. Việc tạo hoặc cập nhật hồ sơ, lưu tài liệu và ghi lịch sử trạng thái được thực hiện trong cùng một database transaction.
    \item \textbf{Nhà tài trợ xét duyệt:} Webadmin lấy danh sách hồ sơ theo tổ chức qua \texttt{GET /student-scholarships/organization/\{organization\_id\}}. Nhà tài trợ xem chi tiết hồ sơ, tài liệu đính kèm, GPA, thông tin học thuật và cập nhật trạng thái qua \texttt{PUT /student-scholarships/\{id\}/status}.
    \item \textbf{Sinh viên theo dõi kết quả:} Sinh viên xem các hồ sơ đã nộp qua \texttt{GET /scholarships/my-scholarships}, xem chi tiết từng hồ sơ qua \texttt{GET /scholarships/my-scholarships/\{id\}}, và có thể hủy hồ sơ khi còn ở trạng thái cho phép qua \texttt{DELETE /scholarships/my-scholarships/\{id\}}.
\end{enumerate}
